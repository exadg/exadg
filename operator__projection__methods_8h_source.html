31/*
32 * Base class for projection-type incompressible Navier-Stokes solvers such as the high-order dual
33 * splitting (velocity-correction) scheme or pressure correction schemes.
34 */
35template<int dim, typename Number>
36class OperatorProjectionMethods : public SpatialOperatorBase<dim, Number>
37{
38protected:
39 typedef SpatialOperatorBase<dim, Number> Base;
40
41 typedef typename Base::VectorType VectorType;
42 typedef typename Base::MultigridPoisson MultigridPoisson;
43
44public:
45 /*
46 * Constructor.
47 */
48 OperatorProjectionMethods(std::shared_ptr<Grid<dim> const> grid,
49 std::shared_ptr<GridMotionInterface<dim, Number>> grid_motion,
50 std::shared_ptr<BoundaryDescriptor<dim> const> boundary_descriptor,
51 std::shared_ptr<FieldFunctions<dim> const> field_functions,
52 Parameters const & parameters,
53 std::string const & field,
54 MPI_Comm const & mpi_comm);
55
56 /*
57 * Destructor.
58 */
59 virtual ~OperatorProjectionMethods();
60
61 /*
62 * Calls setup() function of base class and additionally initializes the pressure Poisson operator
63 * needed for projection-type methods.
64 */
65 void
66 setup(std::shared_ptr<dealii::MatrixFree<dim, Number>> matrix_free,
67 std::shared_ptr<MatrixFreeData<dim, Number>> matrix_free_data,
68 std::string const & dof_index_temperature = "") override;
69
70 void
71 update_after_grid_motion() override;
72
73 /*
74 * This function evaluates the rhs-contribution of the viscous term and adds the result to the
75 * dst-vector.
76 */
77 void
78 do_rhs_add_viscous_term(VectorType & dst, double const time) const;
79
80 /*
81 * Pressure Poisson equation: This function evaluates the inhomogeneous parts of boundary face
82 * integrals of the negative Laplace operator and adds the result to the dst-vector.
83 */
84 void
85 do_rhs_ppe_laplace_add(VectorType & dst, double const & time) const;
86
87 /*
88 * This function solves the pressure Poisson equation and returns the number of iterations.
89 */
90 unsigned int
91 do_solve_pressure(VectorType & dst,
92 VectorType const & src,
93 bool const update_preconditioner) const;
94
95 /*
96 * This function applies the projection operator (used for throughput measurements).
97 */
98 void
99 apply_projection_operator(VectorType & dst, VectorType const & src) const;
100
101 /*
102 * This function applies the Laplace operator (used for throughput measurements).
103 */
104 void
105 apply_laplace_operator(VectorType & dst, VectorType const & src) const;
106
107protected:
108 /*
109 * Initializes the preconditioner and solver for the pressure Poisson equation. Can be done in
110 * this base class since it is the same for dual-splitting and pressure-correction. The function
111 * is declared virtual so that individual initializations required for derived class can be added
112 * where needed.
113 */
114 virtual void
115 setup_pressure_poisson_solver();
116
117 // Pressure Poisson equation (operator, preconditioner, solver).
118 Poisson::LaplaceOperator<dim, Number, 1> laplace_operator;
119
120 std::shared_ptr<PreconditionerBase<Number>> preconditioner_pressure_poisson;
121
122 std::shared_ptr<Krylov::SolverBase<VectorType>> pressure_poisson_solver;
123
124private:
125 /*
126 * Initialization functions called during setup of pressure Poisson solver.
127 */
128 void
129 initialize_laplace_operator();
130
131 void
132 initialize_preconditioner_pressure_poisson();
133
134 void
135 initialize_solver_pressure_poisson();
136};
