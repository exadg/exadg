72 /*
73 * Volume flux, i.e., the term occurring in the volume integral
74 */
75 inline DEAL_II_ALWAYS_INLINE //
76 vector
77 get_volume_flux(Integrator const & integrator,
78 IntegratorScalar const & integrator_temperature,
79 unsigned int const q,
80 Number const & time) const
81 {
82 dealii::Point<dim, scalar> q_points = integrator.quadrature_point(q);
83
84 vector f = FunctionEvaluator<1, dim, Number>::value(data.f, q_points, time);
85
86 if(data.boussinesq_term)
87 {
88 vector g = FunctionEvaluator<1, dim, Number>::value(data.gravitational_force, q_points, time);
89 scalar T = integrator_temperature.get_value(q);
90 scalar T_ref = data.reference_temperature;
91 // solve only for the dynamic pressure variations
92 if(data.boussinesq_dynamic_part_only)
93 f += g * (-data.thermal_expansion_coefficient * (T - T_ref));
94 else // includes hydrostatic component
95 f += g * (1.0 - data.thermal_expansion_coefficient * (T - T_ref));
96 }
97
98 return f;
99 }
100
101private:
102 mutable RHSKernelData<dim> data;
103};
