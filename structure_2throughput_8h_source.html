49void
50create_input_file(std::string const & input_file)
51{
52 dealii::ParameterHandler prm;
53
54 GeneralParameters general;
55 general.add_parameters(prm);
56
57 HypercubeResolutionParameters resolution;
58 resolution.add_parameters(prm);
59
60 ThroughputParameters throughput;
61 throughput.add_parameters(prm);
62
63 try
64 {
65 // we have to assume a default dimension and default Number type
66 // for the automatic generation of a default input file
67 unsigned int const Dim = 2;
68 typedef double Number;
69 Structure::get_application<Dim, Number>(input_file, MPI_COMM_WORLD)->add_parameters(prm);
70 }
71 catch(...)
72 {
73 }
74
75 prm.print_parameters(input_file,
76 dealii::ParameterHandler::Short |
77 dealii::ParameterHandler::KeepDeclarationOrder);
78}

80template<int dim, typename Number>
81void
82run(ThroughputParameters const & throughput,
83 std::string const & input_file,
84 unsigned int const degree,
85 unsigned int const refine_space,
86 unsigned int const n_cells_1d,
87 MPI_Comm const & mpi_comm,
88 bool const is_test)
89{
90 std::shared_ptr<Structure::ApplicationBase<dim, Number>> application =
91 Structure::get_application<dim, Number>(input_file, mpi_comm);
92
93 application->set_parameters_throughput_study(degree, refine_space, n_cells_1d);
94
95 std::shared_ptr<Structure::Driver<dim, Number>> driver =
96 std::make_shared<Structure::Driver<dim, Number>>(mpi_comm, application, is_test, true);
97
98 driver->setup();
99
100 std::tuple<unsigned int, dealii::types::global_dof_index, double> wall_time =
101 driver->apply_operator(throughput.operator_type,
102 throughput.n_repetitions_inner,
103 throughput.n_repetitions_outer);
104
105 throughput.wall_times.push_back(wall_time);
106}
107} // namespace ExaDG

109int
110main(int argc, char ** argv)
111{
112#ifdef EXADG_WITH_LIKWID
113 LIKWID_MARKER_INIT;
114#endif
115
116 dealii::Utilities::MPI::MPI_InitFinalize mpi(argc, argv, 1);
117
118 MPI_Comm mpi_comm(MPI_COMM_WORLD);
119
120 std::string input_file;
121
122 if(argc == 1)
123 {
124 if(dealii::Utilities::MPI::this_mpi_process(mpi_comm) == 0)
125 {
126 // clang-format off
127 std::cout << "To run the program, use: ./throughput input_file" << std::endl
128 << "To setup the input file, use: ./throughput input_file --help" << std::endl;
129 // clang-format on
130 }
131
132 return 0;
133 }
134 else if(argc >= 2)
135 {
136 input_file = std::string(argv[1]);
137
138 if(argc == 3 and std::string(argv[2]) == "--help")
139 {
140 if(dealii::Utilities::MPI::this_mpi_process(mpi_comm) == 0)
141 ExaDG::create_input_file(input_file);
142
143 return 0;
144 }
145 }
146
147 ExaDG::GeneralParameters general(input_file);
148 ExaDG::HypercubeResolutionParameters resolution(input_file, general.dim);
149 ExaDG::ThroughputParameters throughput(input_file);
150
151 // fill resolution vector depending on the operator_type
152 resolution.fill_resolution_vector(&ExaDG::Structure::get_dofs_per_element, input_file);
153
154 // loop over resolutions vector and run simulations
155 for(auto iter = resolution.resolutions.begin(); iter != resolution.resolutions.end(); ++iter)
156 {
157 unsigned int const degree = std::get<0>(*iter);
158 unsigned int const refine_space = std::get<1>(*iter);
159 unsigned int const n_cells_1d = std::get<2>(*iter);
160
161 if(general.dim == 2 and general.precision == "float")
162 ExaDG::run<2, float>(
163 throughput, input_file, degree, refine_space, n_cells_1d, mpi_comm, general.is_test);
164 else if(general.dim == 2 and general.precision == "double")
165 ExaDG::run<2, double>(
166 throughput, input_file, degree, refine_space, n_cells_1d, mpi_comm, general.is_test);
167 else if(general.dim == 3 and general.precision == "float")
168 ExaDG::run<3, float>(
169 throughput, input_file, degree, refine_space, n_cells_1d, mpi_comm, general.is_test);
170 else if(general.dim == 3 and general.precision == "double")
171 ExaDG::run<3, double>(
172 throughput, input_file, degree, refine_space, n_cells_1d, mpi_comm, general.is_test);
173 else
174 AssertThrow(false,
175 dealii::ExcMessage("Only dim = 2|3 and precision = float|double implemented."));
176 }
177
178 if(not(general.is_test))
179 throughput.print_results(mpi_comm);
180
181#ifdef EXADG_WITH_LIKWID
182 LIKWID_MARKER_CLOSE;
183#endif
184
185 return 0;
186}
