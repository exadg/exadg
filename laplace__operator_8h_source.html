45template<int dim, typename Number, int n_components = 1>
46class LaplaceKernel
47{
48private:
49 typedef dealii::LinearAlgebra::distributed::Vector<Number> VectorType;
50
51 typedef dealii::VectorizedArray<Number> scalar;
52
53 typedef FaceIntegrator<dim, n_components, Number> IntegratorFace;

55public:
56 LaplaceKernel() : degree(1), tau(dealii::make_vectorized_array<Number>(0.0))
57 {
58 }
59
60 void
61 reinit(dealii::MatrixFree<dim, Number> const & matrix_free,
62 LaplaceKernelData const & data_in,
63 unsigned int const dof_index)
64 {
65 data = data_in;
66
67 dealii::FiniteElement<dim> const & fe = matrix_free.get_dof_handler(dof_index).get_fe();
68 degree = fe.degree;
69
70 calculate_penalty_parameter(matrix_free, dof_index);
71 }
72
73 void
74 calculate_penalty_parameter(dealii::MatrixFree<dim, Number> const & matrix_free,
75 unsigned int const dof_index)
76 {
77 IP::calculate_penalty_parameter<dim, Number>(array_penalty_parameter, matrix_free, dof_index);
78 }
79
80 IntegratorFlags
81 get_integrator_flags(bool const is_dg) const
82 {
83 IntegratorFlags flags;
84
85 flags.cell_evaluate = dealii::EvaluationFlags::gradients;
86 flags.cell_integrate = dealii::EvaluationFlags::gradients;
87
88 if(is_dg)
89 {
90 flags.face_evaluate = dealii::EvaluationFlags::values | dealii::EvaluationFlags::gradients;
91 flags.face_integrate = dealii::EvaluationFlags::values | dealii::EvaluationFlags::gradients;
92 }
93 else
94 {
95 // evaluation of Neumann BCs for continuous elements
96 flags.face_evaluate = dealii::EvaluationFlags::nothing;
97 flags.face_integrate = dealii::EvaluationFlags::values;
98 }
99
100 return flags;
101 }
102
103 static MappingFlags
104 get_mapping_flags(bool const compute_interior_face_integrals,
105 bool const compute_boundary_face_integrals)
106 {
107 MappingFlags flags;
108
109 flags.cells = dealii::update_gradients | dealii::update_JxW_values;
110
111 if(compute_interior_face_integrals)
112 {
113 flags.inner_faces =
114 dealii::update_gradients | dealii::update_JxW_values | dealii::update_normal_vectors;
115 }
116
117 if(compute_boundary_face_integrals)
118 {
119 flags.boundary_faces = dealii::update_gradients | dealii::update_JxW_values |
120 dealii::update_normal_vectors | dealii::update_quadrature_points;
121 }
122
123 return flags;
124 }
125
126 void
127 reinit_face(IntegratorFace & integrator_m, IntegratorFace & integrator_p) const
128 {
129 tau = std::max(integrator_m.read_cell_data(array_penalty_parameter),
130 integrator_p.read_cell_data(array_penalty_parameter)) *
131 IP::get_penalty_factor<Number>(degree, data.IP_factor);
132 }
133
134 void
135 reinit_boundary_face(IntegratorFace & integrator_m) const
136 {
137 tau = integrator_m.read_cell_data(array_penalty_parameter) *
138 IP::get_penalty_factor<Number>(degree, data.IP_factor);
139 }
140
141 void
142 reinit_face_cell_based(dealii::types::boundary_id const boundary_id,
143 IntegratorFace & integrator_m,
144 IntegratorFace & integrator_p) const
145 {
146 if(boundary_id == dealii::numbers::internal_face_boundary_id) // internal face
147 {
148 tau = std::max(integrator_m.read_cell_data(array_penalty_parameter),
149 integrator_p.read_cell_data(array_penalty_parameter)) *
150 IP::get_penalty_factor<Number>(degree, data.IP_factor);
151 }
152 else // boundary face
153 {
154 tau = integrator_m.read_cell_data(array_penalty_parameter) *
155 IP::get_penalty_factor<Number>(degree, data.IP_factor);
156 }
157 }

159 template<typename T>
160 inline DEAL_II_ALWAYS_INLINE //
161 T
162 calculate_gradient_flux(T const & value_m, T const & value_p) const
163 {
164 return -0.5 * (value_m - value_p);
165 }
166
167 template<typename T>
168 inline DEAL_II_ALWAYS_INLINE //
169 T
170 calculate_value_flux(T const & normal_gradient_m,
171 T const & normal_gradient_p,
172 T const & value_m,
173 T const & value_p) const
174 {
175 return 0.5 * (normal_gradient_m + normal_gradient_p) - tau * (value_m - value_p);
176 }
177
178private:
179 LaplaceKernelData data;
180
181 unsigned int degree;
182
183 dealii::AlignedVector<scalar> array_penalty_parameter;
184
185 mutable scalar tau;
186};
