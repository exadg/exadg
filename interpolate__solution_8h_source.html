33 /*
34  * For a given vector of adjacent cells and points in reference coordinates, determine
35  * and return the dof_indices and shape_values to be used later for interpolation of the
36  * solution.
37  */
38 template<int dim, typename Number>
39 std::vector<std::pair<std::vector<types::global_dof_index>, std::vector<Number>>>
40 get_dof_indices_and_shape_values(
41  std::vector<std::pair<typename Triangulation<dim>::active_cell_iterator, Point<dim>>> const &
42  adjacent_cells,
43  DoFHandler<dim> const & dof_handler,
44  Mapping<dim> const & mapping,
45  LinearAlgebra::distributed::Vector<Number> const & solution)
46 {
47  // fill dof_indices_and_shape_values
48  std::vector<std::pair<std::vector<types::global_dof_index>, std::vector<Number>>>
49  dof_indices_and_shape_values;
50 
51  for(auto cell_tria : adjacent_cells)
52  {
53  // transform Triangulation::active_cell_iterator into DoFHandler::active_cell_iterator,
54  // see constructor of DoFCellAccessor
55  typename DoFHandler<dim>::active_cell_iterator cell_dof = {&dof_handler.get_triangulation(),
56  cell_tria.first->level(),
57  cell_tria.first->index(),
58  &dof_handler};
59 
60  typedef std::pair<typename DoFHandler<dim>::active_cell_iterator, Point<dim>> PairDof;
61  PairDof cell_and_point(cell_dof, cell_tria.second);
62 
63  // go on only if cell is owned by the processor
64  if(cell_and_point.first->is_locally_owned())
65  {
66  const Quadrature<dim> quadrature(
67  GeometryInfo<dim>::project_to_unit_cell(cell_and_point.second));
68 
69  const FiniteElement<dim> & fe = dof_handler.get_fe();
70  FEValues<dim> fe_values(mapping, fe, quadrature, update_values);
71  fe_values.reinit(cell_and_point.first);
72  std::vector<types::global_dof_index> dof_indices(fe.dofs_per_cell);
73  cell_and_point.first->get_dof_indices(dof_indices);
74 
75  std::vector<types::global_dof_index> dof_indices_global(fe.dofs_per_cell);
76  for(unsigned int i = 0; i < fe.dofs_per_cell; ++i)
77  dof_indices_global[i] = solution.get_partitioner()->global_to_local(dof_indices[i]);
78 
79  std::vector<Number> fe_shape_values(fe.dofs_per_cell);
80  for(unsigned int i = 0; i < fe.dofs_per_cell; ++i)
81  fe_shape_values[i] = fe_values.shape_value(i, 0);
82 
83  dof_indices_and_shape_values.emplace_back(dof_indices_global, fe_shape_values);
84  }
85  }
86 
87  return dof_indices_and_shape_values;
88 }

115 /*
116  * The quantity to be evaluated is of type Tensor<0,dim,Number>.
117  */
118 template<int dim, typename Number>
119 struct Interpolator<0, dim, Number>
120 {
121  static inline DEAL_II_ALWAYS_INLINE //
122  Tensor<0, dim, Number>
123  value(DoFHandler<dim> const & dof_handler,
124  LinearAlgebra::distributed::Vector<Number> const & solution,
125  std::vector<types::global_dof_index> const & dof_indices,
126  std::vector<Number> const & fe_shape_values)
127  {
128  Assert(dof_handler.get_fe().dofs_per_cell == fe_shape_values.size(),
129  ExcMessage("Vector fe_shape_values has wrong size."));
130 
131  Number result = Number(0.0);
132 
133  FiniteElement<dim> const & fe = dof_handler.get_fe();
134  for(unsigned int i = 0; i < fe.dofs_per_cell; ++i)
135  result += solution.local_element(dof_indices[i]) * fe_shape_values[i];
136 
137  Tensor<0, dim, Number> result_tensor = result;
138 
139  return result_tensor;
140  }
141 };

143 /*

146 template<int dim, typename Number>
147 struct Interpolator<1, dim, Number>
148 {
149  static inline DEAL_II_ALWAYS_INLINE //
150  Tensor<1, dim, Number>
151  value(DoFHandler<dim> const & dof_handler,
152  LinearAlgebra::distributed::Vector<Number> const & solution,
153  std::vector<types::global_dof_index> const & dof_indices,
154  std::vector<Number> const & fe_shape_values)
155  {
156  Assert(dof_handler.get_fe().dofs_per_cell == fe_shape_values.size(),
157  ExcMessage("Vector fe_shape_values has wrong size."));
158 
159  Tensor<1, dim, Number> result;
160 
161  FiniteElement<dim> const & fe = dof_handler.get_fe();
162  for(unsigned int i = 0; i < fe.dofs_per_cell; ++i)
163  result[fe.system_to_component_index(i).first] +=
164  solution.local_element(dof_indices[i]) * fe_shape_values[i];
165 
166  return result;
167  }
168 };
