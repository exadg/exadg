92 void
93 fill_grid_coordinates_vector(dealii::Mapping<dim> const & mapping,
94 VectorType & grid_coordinates,
95 dealii::DoFHandler<dim> const & dof_handler) const
96 {
97 if(grid_coordinates.size() != dof_handler.n_dofs())
98 {
99 dealii::IndexSet relevant_dofs_grid;
100 dealii::DoFTools::extract_locally_relevant_dofs(dof_handler, relevant_dofs_grid);
101 grid_coordinates.reinit(dof_handler.locally_owned_dofs(),
102 relevant_dofs_grid,
103 dof_handler.get_communicator());
104 }
105 else
106 {
107 grid_coordinates = 0;
108 }
109
110 dealii::FiniteElement<dim> const & fe = dof_handler.get_fe();
111
112 // Set up dealii::FEValues with base element and the Gauss-Lobatto quadrature to
113 // reduce setup cost, as we only use the geometry information (this means
114 // we need to call fe_values.reinit(cell) with Triangulation::cell_iterator
115 // rather than dealii::DoFHandler::cell_iterator).
116 dealii::FE_Nothing<dim> dummy_fe;
117 dealii::FEValues<dim> fe_values(mapping,
118 dummy_fe,
119 dealii::QGaussLobatto<dim>(fe.degree + 1),
120 dealii::update_quadrature_points);
121
122 std::vector<dealii::types::global_dof_index> dof_indices(fe.dofs_per_cell);
123
124 std::vector<std::array<unsigned int, dim>> component_to_system_index(
125 fe.base_element(0).dofs_per_cell);
126
127 if(fe.dofs_per_vertex > 0) // dealii::FE_Q
128 {
129 for(unsigned int i = 0; i < fe.dofs_per_cell; ++i)
130 {
131 component_to_system_index
132 [hierarchic_to_lexicographic_numbering[fe.system_to_component_index(i).second]]
133 [fe.system_to_component_index(i).first] = i;
134 }
135 }
136 else // dealii::FE_DGQ
137 {
138 for(unsigned int i = 0; i < fe.dofs_per_cell; ++i)
139 {
140 component_to_system_index[fe.system_to_component_index(i).second]
141 [fe.system_to_component_index(i).first] = i;
142 }
143 }
144
145 for(auto const & cell : dof_handler.active_cell_iterators())
146 {
147 if(!cell->is_artificial())
148 {
149 fe_values.reinit(typename dealii::Triangulation<dim>::cell_iterator(cell));
150 cell->get_dof_indices(dof_indices);
151 for(unsigned int i = 0; i < fe_values.n_quadrature_points; ++i)
152 {
153 dealii::Point<dim> const point = fe_values.quadrature_point(i);
154 for(unsigned int d = 0; d < dim; ++d)
155 {
156 if(grid_coordinates.get_partitioner()->in_local_range(
157 dof_indices[component_to_system_index[i][d]]))
158 {
159 grid_coordinates(dof_indices[component_to_system_index[i][d]]) = point[d];
160 }
161 }
162 }
163 }
164 }
165
166 grid_coordinates.update_ghost_values();
167 }

181 void
182 initialize_mapping_q_cache(std::shared_ptr<dealii::Mapping<dim> const> mapping,
183 VectorType const & displacement_vector,
184 dealii::DoFHandler<dim> const & dof_handler)
185 {
186 AssertThrow(dealii::MultithreadInfo::n_threads() == 1, dealii::ExcNotImplemented());
187
188 VectorType displacement_vector_ghosted;
189 if(dof_handler.n_dofs() > 0 && displacement_vector.size() == dof_handler.n_dofs())
190 {
191 dealii::IndexSet locally_relevant_dofs;
192 dealii::DoFTools::extract_locally_relevant_dofs(dof_handler, locally_relevant_dofs);
193 displacement_vector_ghosted.reinit(dof_handler.locally_owned_dofs(),
194 locally_relevant_dofs,
195 dof_handler.get_communicator());
196 displacement_vector_ghosted.copy_locally_owned_data_from(displacement_vector);
197 displacement_vector_ghosted.update_ghost_values();
198 }
199
200 std::shared_ptr<dealii::FEValues<dim>> fe_values;
201
202 dealii::FE_Nothing<dim> fe_nothing;
203 if(mapping.get() != 0)
204 {
205 fe_values =
206 std::make_shared<dealii::FEValues<dim>>(*mapping,
207 fe_nothing,
208 dealii::QGaussLobatto<dim>(this->get_degree() + 1),
209 dealii::update_quadrature_points);
210 }
211
212 // update mapping according to mesh deformation described by displacement vector
213 dealii::MappingQCache<dim>::initialize(
214 dof_handler.get_triangulation(),
215 [&](const typename dealii::Triangulation<dim>::cell_iterator & cell_tria)
216 -> std::vector<dealii::Point<dim>> {
217 unsigned int const scalar_dofs_per_cell =
218 dealii::Utilities::pow(this->get_degree() + 1, dim);
219
220 std::vector<dealii::Point<dim>> grid_coordinates(scalar_dofs_per_cell);
221
222 if(mapping.get() != 0)
223 {
224 fe_values->reinit(cell_tria);
225 // extract displacement and add to original position
226 for(unsigned int i = 0; i < scalar_dofs_per_cell; ++i)
227 {
228 grid_coordinates[i] =
229 fe_values->quadrature_point(this->hierarchic_to_lexicographic_numbering[i]);
230 }
231 }
232
233 // if this function is called with an empty dof-vector, this indicates that the
234 // displacements are zero and the points do not have to be moved
235 if(dof_handler.n_dofs() > 0 && displacement_vector.size() > 0 && cell_tria->is_active() &&
236 !cell_tria->is_artificial())
237 {
238 dealii::FiniteElement<dim> const & fe = dof_handler.get_fe();
239 AssertThrow(fe.element_multiplicity(0) == dim,
240 dealii::ExcMessage("Expected finite element with dim components."));
241
242 typename dealii::DoFHandler<dim>::cell_iterator cell(&cell_tria->get_triangulation(),
243 cell_tria->level(),
244 cell_tria->index(),
245 &dof_handler);
246
247 std::vector<dealii::types::global_dof_index> dof_indices(fe.dofs_per_cell);
248 cell->get_dof_indices(dof_indices);
249
250 for(unsigned int i = 0; i < dof_indices.size(); ++i)
251 {
252 std::pair<unsigned int, unsigned int> const id = fe.system_to_component_index(i);
253
254 if(fe.dofs_per_vertex > 0) // dealii::FE_Q
255 {
256 grid_coordinates[id.second][id.first] += displacement_vector_ghosted(dof_indices[i]);
257 }
258 else // dealii::FE_DGQ
259 {
260 grid_coordinates[this->lexicographic_to_hierarchic_numbering[id.second]][id.first] +=
261 displacement_vector_ghosted(dof_indices[i]);
262 }
263 }
264 }
265
266 return grid_coordinates;
267 });
268 }
269
270 std::vector<unsigned int> hierarchic_to_lexicographic_numbering;
271 std::vector<unsigned int> lexicographic_to_hierarchic_numbering;
272};

282template<int dim, typename Number>
283void
284initialize_multigrid(std::shared_ptr<MappingDoFVector<dim, Number>> mapping_multigrid,
285 std::shared_ptr<dealii::MappingQCache<dim> const> & mapping_q_cache,
286 dealii::Triangulation<dim> const & triangulation)
287{
288 AssertThrow(dealii::MultithreadInfo::n_threads() == 1, dealii::ExcNotImplemented());
289
290 typedef dealii::LinearAlgebra::distributed::Vector<Number> VectorType;
291
292 // we have to project the solution onto all coarse levels of the triangulation
293 dealii::MGLevelObject<VectorType> grid_coordinates_all_levels,
294 grid_coordinates_all_levels_ghosted;
295 unsigned int const n_levels = triangulation.n_global_levels();
296 grid_coordinates_all_levels.resize(0, n_levels - 1);
297 grid_coordinates_all_levels_ghosted.resize(0, n_levels - 1);
298
299 dealii::FESystem<dim> fe(dealii::FE_Q<dim>(mapping_q_cache->get_degree()), dim);
300 dealii::DoFHandler<dim> dof_handler(triangulation);
301 dof_handler.distribute_dofs(fe);
302 dof_handler.distribute_mg_dofs();
303 VectorType grid_coordinates_fine_level;
304 mapping_multigrid->fill_grid_coordinates_vector(*mapping_q_cache,
305 grid_coordinates_fine_level,
306 dof_handler);
307
308 dealii::MGTransferMatrixFree<dim, Number> transfer;
309 transfer.build(dof_handler);
310 transfer.interpolate_to_mg(dof_handler, grid_coordinates_all_levels, grid_coordinates_fine_level);
311
312 for(unsigned int level = 0; level < n_levels; level++)
313 {
314 dealii::IndexSet relevant_dofs;
315 dealii::DoFTools::extract_locally_relevant_level_dofs(dof_handler, level, relevant_dofs);
316
317 grid_coordinates_all_levels_ghosted[level].reinit(
318 dof_handler.locally_owned_mg_dofs(level),
319 relevant_dofs,
320 grid_coordinates_all_levels[level].get_mpi_communicator());
321
322 grid_coordinates_all_levels_ghosted[level].copy_locally_owned_data_from(
323 grid_coordinates_all_levels[level]);
324
325 grid_coordinates_all_levels_ghosted[level].update_ghost_values();
326 }
327
328 AssertThrow(fe.element_multiplicity(0) == dim,
329 dealii::ExcMessage("Expected finite element with dim components."));
330
331 // update mapping for all multigrid levels according to grid coordinates described by static
332 // mapping
333 mapping_multigrid->initialize(
334 dof_handler.get_triangulation(),
335 [&](const typename dealii::Triangulation<dim>::cell_iterator & cell_tria)
336 -> std::vector<dealii::Point<dim>> {
337 unsigned int const level = cell_tria->level();
338
339 typename dealii::DoFHandler<dim>::cell_iterator cell(&cell_tria->get_triangulation(),
340 level,
341 cell_tria->index(),
342 &dof_handler);
343
344 unsigned int const scalar_dofs_per_cell = dealii::Utilities::pow(fe.degree + 1, dim);
345
346 std::vector<dealii::Point<dim>> grid_coordinates(scalar_dofs_per_cell);
347
348 if(cell->level_subdomain_id() != dealii::numbers::artificial_subdomain_id)
349 {
350 std::vector<dealii::types::global_dof_index> dof_indices(fe.dofs_per_cell);
351 cell->get_mg_dof_indices(dof_indices);
352
353 for(unsigned int i = 0; i < dof_indices.size(); ++i)
354 {
355 std::pair<unsigned int, unsigned int> const id = fe.system_to_component_index(i);
356
357 if(fe.dofs_per_vertex > 0) // dealii::FE_Q
358 {
359 grid_coordinates[id.second][id.first] =
360 grid_coordinates_all_levels_ghosted[level](dof_indices[i]);
361 }
362 else // dealii::FE_DGQ
363 {
364 grid_coordinates[mapping_multigrid->lexicographic_to_hierarchic_numbering[id.second]]
365 [id.first] = grid_coordinates_all_levels_ghosted[level](dof_indices[i]);
366 }
367 }
368 }
369
370 return grid_coordinates;
371 });
372}

378template<int dim, typename Number>
379void
380initialize_multigrid(
381 std::vector<std::shared_ptr<MappingDoFVector<dim, Number>>> & coarse_grid_mappings,
382 std::shared_ptr<dealii::MappingQCache<dim> const> & mapping_q_cache,
383 std::vector<std::shared_ptr<dealii::Triangulation<dim> const>> & coarse_grid_triangulations)
384{
385 typedef dealii::LinearAlgebra::distributed::Vector<Number> VectorType;
386
387 dealii::FESystem<dim> fe(dealii::FE_Q<dim>(mapping_q_cache->get_degree()), dim);
388 unsigned int const n_h_levels = coarse_grid_triangulations.size();
389 std::vector<dealii::DoFHandler<dim>> coarse_grid_dof_handlers(n_h_levels);
390 std::vector<dealii::AffineConstraints<Number>> coarse_grid_constraints(n_h_levels);
391 for(unsigned int i = 0; i < n_h_levels; ++i)
392 {
393 coarse_grid_dof_handlers[i].reinit(*coarse_grid_triangulations[i]);
394 coarse_grid_dof_handlers[i].distribute_dofs(fe);
395 // constraints are irrelevant for interpolation
396 coarse_grid_constraints[i].close();
397 }
398 dealii::MGLevelObject<dealii::MGTwoLevelTransfer<dim, VectorType>> transfers(0, n_h_levels - 1);
399 for(unsigned int i = 1; i < n_h_levels; ++i)
400 {
401 transfers[i].reinit_geometric_transfer(coarse_grid_dof_handlers[i],
402 coarse_grid_dof_handlers[i - 1],
403 coarse_grid_constraints[i],
404 coarse_grid_constraints[i - 1]);
405 }
406
407 // a function that initializes the dof-vector for a given level and dof_handler
408 const std::function<void(unsigned int const, VectorType &)> initialize_dof_vector =
409 [&](unsigned int const h_level, VectorType & vector) {
410 dealii::IndexSet locally_relevant_dofs;
411 dealii::DoFTools::extract_locally_relevant_dofs(coarse_grid_dof_handlers[h_level],
412 locally_relevant_dofs);
413 vector.reinit(coarse_grid_dof_handlers[h_level].locally_owned_dofs(),
414 locally_relevant_dofs,
415 coarse_grid_dof_handlers[h_level].get_communicator());
416 };
417
418 dealii::MGTransferGlobalCoarsening<dim, VectorType> mg_transfer_global_coarsening(
419 transfers, initialize_dof_vector);
420
421 dealii::MGLevelObject<VectorType> coarse_grid_coordinates(0, n_h_levels - 1);
422
423 coarse_grid_mappings.resize(n_h_levels);
424 coarse_grid_mappings[n_h_levels - 1] =
425 std::make_shared<MappingDoFVector<dim, Number>>(mapping_q_cache->get_degree());
426
427 // get dof-vector with grid coordinates from the finest h-level
428 coarse_grid_mappings[n_h_levels - 1]->fill_grid_coordinates_vector(
429 *mapping_q_cache,
430 coarse_grid_coordinates[n_h_levels - 1],
431 coarse_grid_dof_handlers[n_h_levels - 1]);
432
433 // transfer grid coordinates to coarser h-levels
434 // the dealii::DoFHandler object will not be used for global coarsening
435 dealii::DoFHandler<dim> dof_handler_dummy;
436 VectorType vector_copy(coarse_grid_coordinates[n_h_levels - 1]);
437 mg_transfer_global_coarsening.interpolate_to_mg(dof_handler_dummy,
438 coarse_grid_coordinates,
439 vector_copy);
440
441 // initialize mapping for all h-levels using the dof-vectors with grid coordinates
442 for(unsigned int h_level = 0; h_level < n_h_levels; ++h_level)
443 {
444 coarse_grid_mappings[h_level] =
445 std::make_shared<MappingDoFVector<dim, Number>>(mapping_q_cache->get_degree());
446
447 // coarse_grid_coordinates describes absolute coordinates -> use an uninitialized mapping
448 std::shared_ptr<dealii::Mapping<dim> const> mapping_dummy;
449 coarse_grid_mappings[h_level]->initialize_mapping_q_cache(mapping_dummy,
450 coarse_grid_coordinates[h_level],
451 coarse_grid_dof_handlers[h_level]);
452 }
453}
