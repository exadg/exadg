100 allocate_vectors() final
101 {
102 pde_operator->initialize_dof_vector(solution);
103 pde_operator->initialize_dof_vector(prediction);

116 void
117 initialize_former_multistep_dof_vectors() final
118 {
119 if(start_with_low_order)
120 {
121 if(vec_evaluated_operators.size() > 0)
122 pde_operator->evaluate(vec_evaluated_operators[0], solution, get_time());
123 }
124 else // start with high order
125 {
126 // fill evaluated operators
127 VectorType temp_sol;
128 pde_operator->initialize_dof_vector(temp_sol);
129 for(unsigned int i = 0; i < vec_evaluated_operators.size(); ++i)
130 {
131 pde_operator->prescribe_initial_conditions(temp_sol, get_previous_time(i));
132 pde_operator->evaluate(vec_evaluated_operators[i], temp_sol, get_previous_time(i));
133 }
134 }

142 void
143 do_timestep_predict()
144 {
145 dealii::Timer timer;
146 timer.restart();
147
148 predrict_solution(prediction, solution, vec_evaluated_operators);
149
150 timer_tree->insert({"Timeloop", "Adams-Bashforth-Moulton"}, timer.wall_time());
151 }
152
153 void
154 do_timestep_correct()
155 {
156 dealii::Timer timer;
157 timer.restart();
158
159 // evaluate operator given the predicted solution
160 pde_operator->evaluate(evaluated_operator_np, prediction, get_next_time());
161 // correct solution
162 correct_solution(solution, evaluated_operator_np, vec_evaluated_operators);
163 // correct operator by evaluating operator with correct solution
164 pde_operator->evaluate(evaluated_operator_np, solution, get_next_time());
165
166 // write output
167 if(this->print_solver_info() and not(this->is_test))
168 {
169 pcout << std::endl << "Adams-Bashforth-Moulton:";
170 print_wall_time(pcout, timer.wall_time());
171 }
172
173 timer_tree->insert({"Timeloop", "Adams-Bashforth-Moulton"}, timer.wall_time());
174 }

176 void
177 do_timestep_solve() final

179 do_timestep_predict();
180 do_timestep_correct();
181 }

183 void
184 correct_solution(VectorType & dst,
185 VectorType const & op_np,
186 std::vector<VectorType> const & ops) const
187 {
188 dst.add(static_cast<Number>(get_time_step_size() * am.get_gamma0()), op_np);
189 for(unsigned int i = 0; i < this->am.get_order() - 1; ++i)
190 dst.add(static_cast<Number>(get_time_step_size() * am.get_alpha(i)), ops[i]);
191 }
192
193 void
194 predrict_solution(VectorType & dst,
195 VectorType const & src,
196 std::vector<VectorType> const & ops) const
197 {
198 dst = src;
199 for(unsigned int i = 0; i < this->ab.get_order(); ++i)
200 dst.add(static_cast<Number>(get_time_step_size() * ab.get_alpha(i)), ops[i]);
201 }

203 void
204 prepare_vectors_for_next_timestep() final
205 {
206 if(vec_evaluated_operators.size() > 0)
207 {
208 push_back(vec_evaluated_operators);
209 std::swap(vec_evaluated_operators[0], evaluated_operator_np);
210 }
211 }
