71inline void
72rename_restart_files(std::string const & filename)
73{
74 // backup: rename current restart file into restart.old in case something fails while writing
75 std::string const from = filename;
76 std::string const to = filename + ".old";
77
78 std::ifstream ifile(from.c_str());
79 if((bool)ifile) // rename only if file already exists
80 {
81 int const error = rename(from.c_str(), to.c_str());
82
83 AssertThrow(error == 0, dealii::ExcMessage("Can not rename file: " + from + " -> " + to));
84 }

115template<typename VectorType, typename BoostArchiveType>
116inline void
117read_write_distributed_vector(VectorType & vector, BoostArchiveType & archive)
118{
119 // Print vector norm here only *before* writing.
120 if(std::is_same<BoostArchiveType, boost::archive::text_oarchive>::value or
121 std::is_same<BoostArchiveType, boost::archive::binary_oarchive>::value)
122 {
123 print_vector_l2_norm(vector);
124 }
125
126 // Depending on VectorType, we have to loop over the blocks to
127 // access the local entries via vector.local_element(i).
128 using Number = typename VectorType::value_type;
129 if constexpr(std::is_same<std::remove_cv_t<VectorType>,
130 dealii::LinearAlgebra::distributed::Vector<Number>>::value)
131 {
132 for(unsigned int i = 0; i < vector.locally_owned_size(); ++i)
133 {
134 archive & vector.local_element(i);
135 }
136 }
137 else if constexpr(std::is_same<std::remove_cv_t<VectorType>,
138 dealii::LinearAlgebra::distributed::BlockVector<Number>>::value)
139 {
140 for(unsigned int i = 0; i < vector.n_blocks(); ++i)
141 {
142 for(unsigned int j = 0; j < vector.block(i).locally_owned_size(); ++j)
143 {
144 archive & vector.block(i).local_element(j);
145 }
146 }
147 }
148 else
149 {
150 AssertThrow(false, dealii::ExcMessage("Reading into this VectorType not supported."));
151 }
152
153 // Print vector norm here only *after* reading.
154 if(std::is_same<BoostArchiveType, boost::archive::text_iarchive>::value or
155 std::is_same<BoostArchiveType, boost::archive::binary_iarchive>::value)
156 {
157 print_vector_l2_norm(vector);
158 }
159}

166template<typename VectorType, typename BlockVectorType>
167std::vector<std::vector<VectorType *>>
168get_vectors_per_block(std::vector<BlockVectorType *> const & block_vectors)
169{
170 unsigned int const n_blocks = block_vectors.at(0)->n_blocks();
171 for(unsigned int i = 0; i < block_vectors.size(); ++i)
172 {
173 AssertThrow(block_vectors[i]->n_blocks() == n_blocks,
174 dealii::ExcMessage("Provided number of blocks per "
175 "BlockVector must be equal."));
176 }
177
178 std::vector<std::vector<VectorType *>> vectors_per_block;
179 for(unsigned int i = 0; i < n_blocks; ++i)
180 {
181 std::vector<VectorType *> vectors;
182 for(unsigned int j = 0; j < block_vectors.size(); ++j)
183 {
184 vectors.push_back(&block_vectors[j]->block(i));
185 }
186 vectors_per_block.push_back(vectors);
187 }
188
189 return vectors_per_block;
190}

197template<int dim, typename BlockVectorType>
198std::vector<BlockVectorType>
199get_block_vectors_from_dof_handlers(
200 unsigned int const n_vectors,
201 std::vector<dealii::DoFHandler<dim, dim> const *> const & dof_handlers)
202{
203 unsigned int const n_blocks = dof_handlers.size();
204
205 // Setup first `BlockVector`
206 BlockVectorType block_vector(n_blocks);
207 for(unsigned int i = 0; i < n_blocks; ++i)
208 {
209 block_vector.block(i).reinit(dof_handlers[i]->locally_owned_dofs(),
210 dof_handlers[i]->get_mpi_communicator());
211 }
212 block_vector.collect_sizes();
213
214 std::vector<BlockVectorType> block_vectors(n_vectors, block_vector);
215
216 return block_vectors;
217}

231template<typename VectorType>
232void
233set_ghost_state(std::vector<VectorType *> const & vectors,
234 std::vector<bool> const & had_ghost_elements)
235{
236 AssertThrow(vectors.size() == had_ghost_elements.size(),
237 dealii::ExcMessage("Vector sizes do not match."));
238
239 for(unsigned int i = 0; i < had_ghost_elements.size(); ++i)
240 {
241 if(had_ghost_elements[i])
242 {
243 vectors[i]->update_ghost_values();
244 }
245 else
246 {
247 vectors[i]->zero_out_ghost_values();
248 }
249 }

257template<int dim, typename TriangulationType>
258inline void
259save_coarse_triangulation(std::string const & filename_base,
260 TriangulationType const & triangulation)
261{
262 if constexpr(std::is_same<std::remove_cv_t<TriangulationType>,
263 dealii::parallel::distributed::Triangulation<dim, dim>>::value or
264 std::is_same<std::remove_cv_t<TriangulationType>,
265 dealii::parallel::fullydistributed::Triangulation<dim, dim>>::value)
266 {
267 AssertThrow(false,
268 dealii::ExcMessage("Only TriangulationType::Serial, i.e., "
269 "dealii::Triangulation<dim> supported."));
270 }
271
272 std::string const filename = filename_base + ".coarse_triangulation";
273 MPI_Comm const & mpi_comm = triangulation.get_mpi_communicator();
274 if(dealii::Utilities::MPI::this_mpi_process(mpi_comm) == 0)
275 {
276 // Serialization only creates a single file, move with one process only.
277 rename_restart_files(filename + ".info");
278 rename_restart_files(filename + "_triangulation.data");
279
280 // For `dealii::Triangulation` the triangulation is the same for all processes.
281 triangulation.save(filename);
282 }

294template<int dim, typename VectorType>
295inline void
296store_vectors_in_triangulation_and_serialize(
297 std::string const & filename_base,
298 std::vector<std::vector<VectorType const *>> const & vectors_per_dof_handler,
299 std::vector<dealii::DoFHandler<dim, dim> const *> const & dof_handlers)
300{
301 AssertThrow(vectors_per_dof_handler.size() > 0,
302 dealii::ExcMessage("No vectors to store in triangulation."));
303 AssertThrow(vectors_per_dof_handler.size() == dof_handlers.size(),
304 dealii::ExcMessage("Number of vectors of vectors and DoFHandlers do not match."));
305 auto const & triangulation = dof_handlers.at(0)->get_triangulation();
306
307 // Check if all the `Triangulation(s)` associated with the DoFHandlers point to the same object.
308 for(unsigned int i = 1; i < dof_handlers.size(); ++i)
309 {
310 AssertThrow(&dof_handlers[i]->get_triangulation() == &triangulation,
311 dealii::ExcMessage("Triangulations of DoFHandlers are not identical."));
312 }
313
314 // Loop over the `DoFHandlers` and store the vectors in the triangulation.
315 std::vector<std::shared_ptr<dealii::SolutionTransfer<dim, VectorType>>> solution_transfers;
316 std::vector<std::vector<bool>> has_ghost_elements_per_dof_handler;
317 for(unsigned int i = 0; i < dof_handlers.size(); ++i)
318 {
319 // Store ghost state.
320 std::vector<bool> has_ghost_elements = get_ghost_state(vectors_per_dof_handler[i]);
321 has_ghost_elements_per_dof_handler.push_back(has_ghost_elements);
322 for(unsigned int j = 0; j < vectors_per_dof_handler[i].size(); ++j)
323 {
324 vectors_per_dof_handler[i][j]->update_ghost_values();
325 print_vector_l2_norm(*vectors_per_dof_handler[i][j]);
326 }
327
328 solution_transfers.push_back(
329 std::make_shared<dealii::SolutionTransfer<dim, VectorType>>(*dof_handlers[i]));
330 solution_transfers[i]->prepare_for_serialization(vectors_per_dof_handler[i]);
331 }
332
333 // Serialize the triangulation keeping a maximum of two snapshots.
334 std::string const filename = filename_base + ".triangulation";
335 MPI_Comm const & mpi_comm = dof_handlers[0]->get_mpi_communicator();
336 if(dealii::Utilities::MPI::this_mpi_process(mpi_comm) == 0)
337 {
338 // Serialization only creates a single file, move with one process only.
339 rename_restart_files(filename);
340 rename_restart_files(filename + ".info");
341 rename_restart_files(filename + "_fixed.data");
342 rename_restart_files(filename + "_triangulation.data");
343 }
344
345 // Collective call for serialization, general case requires ghosted vectors.
346 triangulation.save(filename);
347
348 // Recover ghost state.
349 for(unsigned int i = 0; i < dof_handlers.size(); ++i)
350 {
351 set_ghost_state(vectors_per_dof_handler[i], has_ghost_elements_per_dof_handler[i]);
352 }
353}

360template<int dim, typename VectorType>
361inline void
362store_vectors_in_triangulation_and_serialize(
363 std::string const & filename_base,
364 std::vector<std::vector<VectorType const *>> const & vectors_per_dof_handler,
365 std::vector<dealii::DoFHandler<dim, dim> const *> const & dof_handlers,
366 dealii::Mapping<dim> const & mapping,
367 dealii::DoFHandler<dim> const * dof_handler_mapping,
368 unsigned int const mapping_degree)
369{
370 AssertThrow(vectors_per_dof_handler.size() > 0,
371 dealii::ExcMessage("No vectors to store in triangulation."));
372 AssertThrow(vectors_per_dof_handler.size() == dof_handlers.size(),
373 dealii::ExcMessage("Number of vectors of vectors and DoFHandlers do not match."));
374
375 auto const & triangulation = dof_handlers.at(0)->get_triangulation();
376
377 // Check if all the `Triangulation(s)` associated with the `DoFHandlers` point to the same object.
378 for(unsigned int i = 1; i < dof_handlers.size(); ++i)
379 {
380 AssertThrow(&dof_handlers[i]->get_triangulation() == &triangulation,
381 dealii::ExcMessage("Triangulations of DoFHandlers are not identical."));
382 }
383
384 AssertThrow(triangulation.all_reference_cells_are_hyper_cube(),
385 dealii::ExcMessage("Serialization including mapping not "
386 "supported for non-hypercube cell types."));
387
388 // Initialize vector to hold grid coordinates.
389 bool vector_initialized = false;
390 VectorType vector_grid_coordinates;
391 for(unsigned int i = 0; i < dof_handlers.size(); ++i)
392 {
393 if(dof_handlers[i] == dof_handler_mapping and not vector_initialized)
394 {
395 // Cheaper setup if we already have a vector given in the input arguments.
396 vector_grid_coordinates.reinit(*vectors_per_dof_handler[i][0],
397 true /* omit_zeroing_entries */);
398 vector_initialized = true;
399 break;
400 }
401 }
402
403 if(not vector_initialized)
404 {
405 // More expensive setup extracting the `dealii::IndexSet`.
406 dealii::IndexSet const & locally_owned_dofs = dof_handler_mapping->locally_owned_dofs();
407 dealii::IndexSet const locally_relevant_dofs =
408 dealii::DoFTools::extract_locally_relevant_dofs(*dof_handler_mapping);
409 vector_grid_coordinates.reinit(locally_owned_dofs,
410 locally_relevant_dofs,
411 dof_handler_mapping->get_mpi_communicator());
412 }
413
414 // Fill vector with mapping.
415 MappingDoFVector<dim, typename VectorType::value_type> mapping_dof_vector(mapping_degree);
416 mapping_dof_vector.fill_grid_coordinates_vector(mapping,
417 vector_grid_coordinates,
418 *dof_handler_mapping);
419
420 // Attach vector holding mapping and corresponding `dof_handler_mapping`.
421 std::vector<std::vector<VectorType const *>> vectors_per_dof_handler_extended =
422 vectors_per_dof_handler;
423 std::vector<VectorType const *> tmp = {&vector_grid_coordinates};
424 vectors_per_dof_handler_extended.push_back(tmp);
425
426 std::vector<dealii::DoFHandler<dim, dim> const *> dof_handlers_extended = dof_handlers;
427 dof_handlers_extended.push_back(dof_handler_mapping);
428
429 // Use utility function that ignores the mapping.
430 store_vectors_in_triangulation_and_serialize(filename_base,
431 vectors_per_dof_handler_extended,
432 dof_handlers_extended);
433}

438template<int dim>
439inline std::shared_ptr<dealii::Triangulation<dim>>
440deserialize_triangulation(std::string const & filename_base,
441 TriangulationType const triangulation_type,
442 MPI_Comm const & mpi_communicator)
443{
444 std::shared_ptr<dealii::Triangulation<dim>> triangulation_old;
445
446 // Deserialize the checkpointed triangulation,
447 if(triangulation_type == TriangulationType::Serial)
448 {
449 triangulation_old = std::make_shared<dealii::Triangulation<dim>>();
450 triangulation_old->load(filename_base + ".triangulation");
451 }
452 else if(triangulation_type == TriangulationType::Distributed)
453 {
454 // Deserialize the coarse triangulation to be stored by the user
455 // during `create_grid` in the respective application.
456 dealii::Triangulation<dim, dim> coarse_triangulation;
457 try
458 {
459 coarse_triangulation.load(filename_base + ".coarse_triangulation");
460 }
461 catch(...)
462 {
463 AssertThrow(false,
464 dealii::ExcMessage(
465 "Deserializing coarse triangulation expected in\n" + filename_base +
466 ".coarse_triangulation\n"
467 "make sure to store the coarse grid during `create_grid`\n"
468 "in the respective application.h using TriangulationType::Serial."));
469 }
470
471 std::shared_ptr<dealii::parallel::distributed::Triangulation<dim>> tmp =
472 std::make_shared<dealii::parallel::distributed::Triangulation<dim>>(mpi_communicator);
473
474 tmp->copy_triangulation(coarse_triangulation);
475 coarse_triangulation.clear();
476
477 // We do not need manifolds when applying the refinements, since we recover the mapping
478 // separately.
479 tmp->reset_all_manifolds();
480 tmp->set_all_manifold_ids(dealii::numbers::flat_manifold_id);
481 tmp->load(filename_base + ".triangulation");
482
483 triangulation_old = std::dynamic_pointer_cast<dealii::Triangulation<dim>>(tmp);
484 }
485 else if(triangulation_type == TriangulationType::FullyDistributed)
486 {
487 // Note that the number of MPI processes the triangulation was
488 // saved with cannot change and hence autopartitioning is disabled.
489 std::shared_ptr<dealii::parallel::fullydistributed::Triangulation<dim>> tmp =
490 std::make_shared<dealii::parallel::fullydistributed::Triangulation<dim>>(mpi_communicator);
491 tmp->load(filename_base + ".triangulation");
492
493 triangulation_old = std::dynamic_pointer_cast<dealii::Triangulation<dim>>(tmp);
494 }
495 else
496 {
497 AssertThrow(false, dealii::ExcMessage("TriangulationType not supported."));
498 }
499
500 return triangulation_old;
501}

508template<int dim, typename VectorType>
509inline void
510load_vectors(std::vector<std::vector<VectorType *>> & vectors_per_dof_handler,
511 std::vector<dealii::DoFHandler<dim, dim> const *> const & dof_handlers)
512{
513 // The `DoFHandlers and vectors` are already initialized and
514 // `vectors_per_dof_handler` contain only owned DoFs.
515 AssertThrow(vectors_per_dof_handler.size() > 0,
516 dealii::ExcMessage("No vectors to load into from triangulation."));
517 AssertThrow(vectors_per_dof_handler.size() == dof_handlers.size(),
518 dealii::ExcMessage("Number of vectors of vectors and DoFHandlers do not match."));
519
520 // Loop over the DoFHandlers and load the vectors stored in
521 // the triangulation the DoFHandlers were initialized with.
522 for(unsigned int i = 0; i < dof_handlers.size(); ++i)
523 {
524 dealii::SolutionTransfer<dim, VectorType> solution_transfer(*dof_handlers[i]);
525
526 // Reinit vectors that do not already have ghost entries.
527 bool all_ghosted = false;
528 for(unsigned int j = 0; j < vectors_per_dof_handler[i].size(); ++j)
529 {
530 if(not vectors_per_dof_handler[i][j]->has_ghost_elements())
531 {
532 all_ghosted = false;
533 break;
534 }
535 }
536 if(not all_ghosted)
537 {
538 dealii::IndexSet const & locally_owned_dofs = dof_handlers[i]->locally_owned_dofs();
539 dealii::IndexSet const locally_relevant_dofs =
540 dealii::DoFTools::extract_locally_relevant_dofs(*dof_handlers[i]);
541 for(unsigned int j = 0; j < vectors_per_dof_handler[i].size(); ++j)
542 {
543 if(not vectors_per_dof_handler[i][j]->has_ghost_elements())
544 {
545 vectors_per_dof_handler[i][j]->reinit(locally_owned_dofs,
546 locally_relevant_dofs,
547 dof_handlers[i]->get_mpi_communicator());
548 }
549 }
550 }
551
552 solution_transfer.deserialize(vectors_per_dof_handler[i]);
553
554 for(unsigned int j = 0; j < vectors_per_dof_handler[i].size(); ++j)
555 {
556 print_vector_l2_norm(*vectors_per_dof_handler[i][j]);
557 }
558 }
559}

565template<int dim, typename VectorType>
566inline std::shared_ptr<MappingDoFVector<dim, typename VectorType::value_type>>
567load_vectors(std::vector<std::vector<VectorType *>> & vectors_per_dof_handler,
568 std::vector<dealii::DoFHandler<dim, dim> const *> const & dof_handlers,
569 dealii::DoFHandler<dim> const * dof_handler_mapping,
570 unsigned int const mapping_degree)
571{
572 // We need a collective call to `SolutionTransfer::deserialize()` with all vectors in a
573 // single container. Hence, create a mapping vector and add a pointer to the input argument.
574 dealii::IndexSet const & locally_owned_dofs = dof_handler_mapping->locally_owned_dofs();
575 VectorType vector_grid_coordinates(locally_owned_dofs,
576 dof_handler_mapping->get_mpi_communicator());
577
578 // Standard utility function, sequence as in `store_vectors_in_triangulation_and_serialize()`.
579 std::vector<std::vector<VectorType *>> vectors_per_dof_handler_extended = vectors_per_dof_handler;
580 std::vector<VectorType *> tmp = {&vector_grid_coordinates};
581 vectors_per_dof_handler_extended.push_back(tmp);
582 std::vector<dealii::DoFHandler<dim, dim> const *> dof_handlers_extended = dof_handlers;
583 dof_handlers_extended.push_back(dof_handler_mapping);
584
585 load_vectors(vectors_per_dof_handler_extended, dof_handlers_extended);
586
587 // Reconstruct the mapping given the deserialized grid coordinate vector.
588 std::shared_ptr<MappingDoFVector<dim, typename VectorType::value_type>> mapping_dof_vector =
589 std::make_shared<MappingDoFVector<dim, typename VectorType::value_type>>(mapping_degree);
590 mapping_dof_vector->initialize_mapping_from_dof_vector(nullptr /* mapping */,
591 vector_grid_coordinates,
592 *dof_handler_mapping);
593 return mapping_dof_vector;
594}
