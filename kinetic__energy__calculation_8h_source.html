83template<int dim, typename Number>
84class KineticEnergyCalculator
85{
86public:
87 static unsigned int const number_vorticity_components = (dim == 2) ? 1 : dim;
88
89 typedef dealii::LinearAlgebra::distributed::Vector<Number> VectorType;
90
91 typedef dealii::VectorizedArray<Number> scalar;
92 typedef dealii::Tensor<1, dim, dealii::VectorizedArray<Number>> vector;
93 typedef dealii::Tensor<2, dim, dealii::VectorizedArray<Number>> tensor;
94
95 KineticEnergyCalculator(MPI_Comm const & comm);
96
97 void
98 setup(dealii::MatrixFree<dim, Number> const & matrix_free_in,
99 unsigned int const dof_index_in,
100 unsigned int const quad_index_in,
101 KineticEnergyData const & kinetic_energy_data_in);
102
103 void
104 evaluate(VectorType const & velocity, double const & time, int const & time_step_number);
105
106protected:
107 void
108 calculate_basic(VectorType const & velocity,
109 double const time,
110 unsigned int const time_step_number);
111
112 /*
113 * This function calculates the kinetic energy
114 *
115 * Kinetic energy: E_k = 1/V * 1/2 * (1,u*u)_Omega, V=(1,1)_Omega is the volume
116 *
117 * Enstrophy: 1/V * 0.5 (1,rot(u)*rot(u))_Omega, V=(1,1)_Omega is the volume
118 *
119 * Dissipation rate: epsilon = nu/V * (1, grad(u):grad(u))_Omega, V=(1,1)_Omega is the volume
120 *
121 * Note that
122 *
123 * epsilon = 2 * nu * Enstrophy
124 *
125 * for incompressible flows (div(u)=0) and periodic boundary conditions.
126 */
127 Number
128 integrate(dealii::MatrixFree<dim, Number> const & matrix_free_data,
129 VectorType const & velocity,
130 Number & energy,
131 Number & enstrophy,
132 Number & dissipation,
133 Number & max_vorticity);
134
135 void
136 cell_loop(dealii::MatrixFree<dim, Number> const & data,
137 std::vector<Number> & dst,
138 VectorType const & src,
139 std::pair<unsigned int, unsigned int> const & cell_range);
140
141 MPI_Comm const mpi_comm;
142
143 bool clear_files;
144
145 dealii::MatrixFree<dim, Number> const * matrix_free;
146 unsigned int dof_index, quad_index;
147 KineticEnergyData data;
148};
