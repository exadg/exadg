34// compute exterior pressure values for different boundary types
35template<int dim, typename Number>
36inline DEAL_II_ALWAYS_INLINE //
37 dealii::VectorizedArray<Number>
38 calculate_exterior_value_pressure(unsigned int const q,
39 FaceIntegrator<dim, 1, Number> const & integrator_m,
40 BoundaryTypeP const & boundary_type,
41 dealii::types::boundary_id const boundary_id,
42 BoundaryDescriptorP<dim> const & boundary_descriptor,
43 Number const time)
44{
45 if(boundary_type == BoundaryTypeP::Dirichlet)
46 {
47 auto const g = FunctionEvaluator<0, dim, Number>::value(
48 *boundary_descriptor.dirichlet_bc.find(boundary_id)->second,
49 integrator_m.quadrature_point(q),
50 time);
51 return -integrator_m.get_value(q) + Number{2.0} * g;
52 }
53 else
54 {
55 AssertThrow(false, dealii::ExcMessage("Boundary type of face is invalid or not implemented."));
56 }
57}

59// compute exterior velocity values for different boundary types
60template<int dim, typename Number>
61inline DEAL_II_ALWAYS_INLINE //
62 dealii::Tensor<1, dim, dealii::VectorizedArray<Number>>
63 calculate_exterior_value_velocity(unsigned int const q,
64 FaceIntegrator<dim, dim, Number> const & integrator_m,
65 BoundaryTypeU const & boundary_type,
66 dealii::types::boundary_id const boundary_id,
67 BoundaryDescriptorU<dim> const & boundary_descriptor,
68 Number const time)
69{
70 (void)boundary_id;
71 (void)boundary_descriptor;
72 (void)time;
73
74 if(boundary_type == BoundaryTypeU::Neumann)
75 {
76 return integrator_m.get_value(q);
77 }
78 else
79 {
80 AssertThrow(false, dealii::ExcMessage("Boundary type of face is invalid or not implemented."));
81 }
82}

88template<int dim, typename Number>
89class BoundaryFaceIntegratorP : public BoundaryFaceIntegratorBase<BoundaryDescriptorP<dim>, Number>
90{
91 using FaceIntegratorP = FaceIntegrator<dim, 1, Number>;
92
93public:
94 BoundaryFaceIntegratorP(FaceIntegratorP const & integrator_m_in,
95 BoundaryDescriptorP<dim> const & boundary_descriptor_in)
96 : BoundaryFaceIntegratorBase<BoundaryDescriptorP<dim>, Number>(
97 integrator_m_in.get_matrix_free(),

101 }
102
103 inline DEAL_II_ALWAYS_INLINE //

115private:
