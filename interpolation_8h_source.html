174  interpolate(const double *& src)
175  {
176  // allocate dst- and src-vector
177  AlignedVector<double> temp1(MAX(dofs_source, dofs_target));
178  AlignedVector<double> temp2(MAX(dofs_source, dofs_target));
179 
180  // get start point of arrays
181  const double * src_ = src;
182  double * dst_ = dst;
183  // loop over all cells
184  for(int c = 0; c < cells; c++)
185  {
186  // loop over all velocity directions
187  for(int d = 0; d < DIM; d++)
188  {
189  // perform interpolation
190  if(DIM == 2)
191  {
192  // ... 2D
193  dealii::internal::
194  EvaluatorTensorProduct<dealii::internal::evaluate_general, 2, 0, 0, double>
195  eval_val(shape_values, shape_values, shape_values, points_source, points_target);
196  eval_val.template values<0, true, false>(src_, temp2.begin());
197  eval_val.template values<1, true, false>(temp2.begin(), temp1.begin());
198  }
199  else
200  {
201  // ... 3D
202  dealii::internal::
203  EvaluatorTensorProduct<dealii::internal::evaluate_general, 3, 0, 0, double>
204  eval_val(shape_values, shape_values, shape_values, points_source, points_target);
205  eval_val.template values<0, true, false>(src_, temp1.begin());
206  eval_val.template values<1, true, false>(temp1.begin(), temp2.begin());
207  eval_val.template values<2, true, false>(temp2.begin(), temp1.begin());
208  }
209  // write interpolated data permuted into destination vector
210  for(int i = 0; i < dofs_target; i++)
211  dst_[i * DIM + d] = temp1[i];
212  // go to next dimension
213  src_ += dofs_source;
214  }
215  // go to next cell
216  dst_ += dofs_target * DIM;
217  }
218  }
