37 /*
38  * This function wraps the copy of a PETSc object (sparse matrix,
39  * preconditioner) with a dealii::LinearAlgebra::distributed::Vector, taking
40  * pre-allocated PETSc vector objects (with struct name `Vec`, aka
41  * VectorTypePETSc) for the temporary operations
42  */
43 template<typename VectorType>
44 void
45 apply_petsc_operation(VectorType & dst,
46  VectorType const & src,
47  VectorTypePETSc & petsc_vector_dst,
48  VectorTypePETSc & petsc_vector_src,
49  std::function<void(PETScWrappers::VectorBase &,
50  PETScWrappers::VectorBase const &)> petsc_operation)
51 {
52  {
53  // copy to PETSc internal vector type because there is currently no such
54  // function in deal.II (and the transition via ReadWriteVector is too
55  // slow/poorly tested)
56  PetscInt begin, end;
57  PetscErrorCode ierr = VecGetOwnershipRange(petsc_vector_src, &begin, &end);
58  AssertThrow(ierr == 0, ExcPETScError(ierr));
59 
60  PetscScalar * ptr;
61  ierr = VecGetArray(petsc_vector_src, &ptr);
62  AssertThrow(ierr == 0, ExcPETScError(ierr));
63 
64  const PetscInt local_size = src.get_partitioner()->locally_owned_size();
65  AssertDimension(local_size, static_cast<unsigned int>(end - begin));
66  for(PetscInt i = 0; i < local_size; ++i)
67  {
68  ptr[i] = src.local_element(i);
69  }
70 
71  ierr = VecRestoreArray(petsc_vector_src, &ptr);
72  AssertThrow(ierr == 0, ExcPETScError(ierr));
73  }
74 
75  // wrap `Vec` (aka VectorTypePETSc) into VectorBase (without copying data)
76  PETScWrappers::VectorBase petsc_dst(petsc_vector_dst);
77  PETScWrappers::VectorBase petsc_src(petsc_vector_src);
78 
79  petsc_operation(petsc_dst, petsc_src);
80 
81  {
82  PetscInt begin, end;
83  PetscErrorCode ierr = VecGetOwnershipRange(petsc_vector_dst, &begin, &end);
84  AssertThrow(ierr == 0, ExcPETScError(ierr));
85 
86  PetscScalar * ptr;
87  ierr = VecGetArray(petsc_vector_dst, &ptr);
88  AssertThrow(ierr == 0, ExcPETScError(ierr));
89 
90  const PetscInt local_size = dst.get_partitioner()->locally_owned_size();
91  AssertDimension(local_size, static_cast<unsigned int>(end - begin));
92 
93  for(PetscInt i = 0; i < local_size; ++i)
94  {
95  dst.local_element(i) = ptr[i];
96  }
97 
98  ierr = VecRestoreArray(petsc_vector_dst, &ptr);
99  AssertThrow(ierr == 0, ExcPETScError(ierr));
100  }
101 }

103 /*
104  * This function wraps the copy of a PETSc object (sparse matrix,
105  * preconditioner) with a dealii::LinearAlgebra::distributed::Vector,
106  * allocating a PETSc vectors and then calling the other function
107  */
108 template<typename VectorType>
109 void
110 apply_petsc_operation(VectorType & dst,
111  VectorType const & src,
112  MPI_Comm const & petsc_mpi_communicator,
113  std::function<void(PETScWrappers::VectorBase &,
114  PETScWrappers::VectorBase const &)> petsc_operation)
115 {
116  VectorTypePETSc petsc_vector_dst, petsc_vector_src;
117  VecCreateMPI(petsc_mpi_communicator,
118  dst.get_partitioner()->locally_owned_size(),
119  PETSC_DETERMINE,
120  &petsc_vector_dst);
121  VecCreateMPI(petsc_mpi_communicator,
122  src.get_partitioner()->locally_owned_size(),
123  PETSC_DETERMINE,
124  &petsc_vector_src);
125 
126  apply_petsc_operation(dst, src, petsc_vector_dst, petsc_vector_src, petsc_operation);
127 
128  PetscErrorCode ierr = VecDestroy(&petsc_vector_dst);
129  AssertThrow(ierr == 0, ExcPETScError(ierr));
130  ierr = VecDestroy(&petsc_vector_src);
131  AssertThrow(ierr == 0, ExcPETScError(ierr));
132 }
