77  /*
78  * This function extracts the grid coordinates of the current mesh configuration, i.e.,
79  * a mapping describing the mesh displacement has to be used here.
80  */
81  void
82  fill_grid_coordinates_vector(VectorType & vector, DoFHandler<dim> const & dof_handler)
83  {
84  Mapping<dim> const & mapping = *mapping_ale;
85 
86  if(vector.size() != dof_handler.n_dofs())
87  {
88  IndexSet relevant_dofs_grid;
89  DoFTools::extract_locally_relevant_dofs(dof_handler, relevant_dofs_grid);
90  vector.reinit(dof_handler.locally_owned_dofs(), relevant_dofs_grid, mpi_comm);
91  }
92  else
93  vector = 0;
94 
95  FiniteElement<dim> const & fe = dof_handler.get_fe();
96 
97  // Set up FEValues with base element and the Gauss-Lobatto quadrature to
98  // reduce setup cost, as we only use the geometry information (this means
99  // we need to call fe_values.reinit(cell) with Triangulation::cell_iterator
100  // rather than DoFHandler::cell_iterator).
101  FE_Nothing<dim> dummy_fe;
102  FEValues<dim> fe_values(mapping,
103  dummy_fe,
104  QGaussLobatto<dim>(fe.degree + 1),
105  update_quadrature_points);
106 
107  std::vector<types::global_dof_index> dof_indices(fe.dofs_per_cell);
108 
109  std::vector<std::array<unsigned int, dim>> component_to_system_index(
110  fe.base_element(0).dofs_per_cell);
111 
112  if(fe.dofs_per_vertex > 0) // FE_Q
113  {
114  for(unsigned int i = 0; i < fe.dofs_per_cell; ++i)
115  {
116  component_to_system_index
117  [hierarchic_to_lexicographic_numbering[fe.system_to_component_index(i).second]]
118  [fe.system_to_component_index(i).first] = i;
119  }
120  }
121  else // FE_DGQ
122  {
123  for(unsigned int i = 0; i < fe.dofs_per_cell; ++i)
124  {
125  component_to_system_index[fe.system_to_component_index(i).second]
126  [fe.system_to_component_index(i).first] = i;
127  }
128  }
129 
130  for(auto const & cell : dof_handler.active_cell_iterators())
131  {
132  if(!cell->is_artificial())
133  {
134  fe_values.reinit(typename Triangulation<dim>::cell_iterator(cell));
135  cell->get_dof_indices(dof_indices);
136  for(unsigned int i = 0; i < fe_values.n_quadrature_points; ++i)
137  {
138  Point<dim> const point = fe_values.quadrature_point(i);
139  for(unsigned int d = 0; d < dim; ++d)
140  if(vector.get_partitioner()->in_local_range(
141  dof_indices[component_to_system_index[i][d]]))
142  vector(dof_indices[component_to_system_index[i][d]]) = point[d];
143  }
144  }
145  }
146 
147  vector.update_ghost_values();
148  }

156 protected:
157  void
158  initialize_mapping_q_cache(Mapping<dim> const & mapping,
159  Triangulation<dim> const & triangulation,
160  std::shared_ptr<Function<dim>> displacement_function)
161  {
162  // dummy FE for compatibility with interface of FEValues
163  FE_Nothing<dim> dummy_fe;
164  FEValues<dim> fe_values(mapping,
165  dummy_fe,
166  QGaussLobatto<dim>(this->mapping_ale->get_degree() + 1),
167  update_quadrature_points);
168 
169  AssertThrow(MultithreadInfo::n_threads() == 1, ExcNotImplemented());
170 
171  this->mapping_ale->initialize(
172  triangulation,
173  [&](typename Triangulation<dim>::cell_iterator const & cell) -> std::vector<Point<dim>> {
174  fe_values.reinit(cell);
175 
176  // compute displacement and add to original position
177  std::vector<Point<dim>> points_moved(fe_values.n_quadrature_points);
178  for(unsigned int i = 0; i < fe_values.n_quadrature_points; ++i)
179  {
180  // need to adjust for hierarchic numbering of MappingQCache
181  Point<dim> const point =
182  fe_values.quadrature_point(this->hierarchic_to_lexicographic_numbering[i]);
183  Point<dim> displacement;
184  for(unsigned int d = 0; d < dim; ++d)
185  displacement[d] = displacement_function->value(point, d);
186 
187  points_moved[i] = point + displacement;
188  }
189 
190  return points_moved;
191  });
192  }

194  void
195  initialize_mapping_q_cache(Mapping<dim> const & mapping,
196  DoFHandler<dim> const & dof_handler,
197  VectorType const & dof_vector)
198  {
199  // we have to project the solution onto all coarse levels of the triangulation
200  // (required for initialization of MappingQCache)
201  MGLevelObject<VectorType> dof_vector_all_levels;
202  unsigned int const n_levels = dof_handler.get_triangulation().n_global_levels();
203  dof_vector_all_levels.resize(0, n_levels - 1);
204 
205  MGTransferMatrixFree<dim, Number> transfer;
206  transfer.build(dof_handler);
207  transfer.interpolate_to_mg(dof_handler, dof_vector_all_levels, dof_vector);
208  for(unsigned int level = 0; level < n_levels; level++)
209  dof_vector_all_levels[level].update_ghost_values();
210 
211  FiniteElement<dim> const & fe = dof_handler.get_fe();
212  AssertThrow(fe.element_multiplicity(0) == dim,
213  ExcMessage("Expected finite element with dim components."));
214 
215  FE_Nothing<dim> dummy_fe;
216  FEValues<dim> fe_values(mapping,
217  dummy_fe,
218  QGaussLobatto<dim>(fe.degree + 1),
219  update_quadrature_points);
220 
221  // update mapping according to mesh deformation computed above
222  this->mapping_ale->initialize(
223  dof_handler.get_triangulation(),
224  [&](const typename Triangulation<dim>::cell_iterator & cell_tria) -> std::vector<Point<dim>> {
225  unsigned int const level = cell_tria->level();
226 
227  typename DoFHandler<dim>::cell_iterator cell(&cell_tria->get_triangulation(),
228  level,
229  cell_tria->index(),
230  &dof_handler);
231 
232  unsigned int const scalar_dofs_per_cell = Utilities::pow(fe.degree + 1, dim);
233 
234  std::vector<Point<dim>> points_moved(scalar_dofs_per_cell);
235 
236  if(cell->level_subdomain_id() != numbers::artificial_subdomain_id)
237  {
238  fe_values.reinit(typename Triangulation<dim>::cell_iterator(cell));
239  std::vector<types::global_dof_index> dof_indices(fe.dofs_per_cell);
240  cell->get_mg_dof_indices(dof_indices);
241 
242  // extract displacement and add to original position
243  for(unsigned int i = 0; i < scalar_dofs_per_cell; ++i)
244  {
245  points_moved[i] =
246  fe_values.quadrature_point(this->hierarchic_to_lexicographic_numbering[i]);
247  }
248 
249  for(unsigned int i = 0; i < dof_indices.size(); ++i)
250  {
251  std::pair<unsigned int, unsigned int> const id = fe.system_to_component_index(i);
252 
253  if(fe.dofs_per_vertex > 0) // FE_Q
254  points_moved[id.second][id.first] += dof_vector_all_levels[level](dof_indices[i]);
255  else // FE_DGQ
256  points_moved[this->lexicographic_to_hierarchic_numbering[id.second]][id.first] +=
257  dof_vector_all_levels[level](dof_indices[i]);
258  }
259  }
260 
261  return points_moved;
262  });
263  }
