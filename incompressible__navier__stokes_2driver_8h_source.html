44// Note: Make sure that the correct time integration scheme is selected in the input file that is
45// compatible with the OperatorType specified here. This also includes the treatment of the
46// convective term (explicit/implicit), e.g., specifying VelocityConvDiffOperator together
47// with an explicit treatment of the convective term will only apply the Helmholtz-like
48// operator.
49
50// clang-format off
51enum class OperatorType{
52 CoupledNonlinearResidual, // nonlinear residual of coupled system of equations
53 CoupledLinearized, // linearized system of equations for coupled solution approach
54 PressurePoissonOperator, // negative Laplace operator (scalar quantity, pressure)
55 ConvectiveOperator, // convective term (vectorial quantity, velocity)
56 HelmholtzOperator, // mass + viscous (vectorial quantity, velocity)
57 ProjectionOperator, // mass + divergence penalty + continuity penalty (vectorial quantity, velocity)
58 VelocityConvDiffOperator, // mass + convective + viscous (vectorial quantity, velocity)
59 InverseMassOperator // inverse mass operator (vectorial quantity, velocity)
60};
61// clang-format on
62
63inline std::string
64enum_to_string(OperatorType const enum_type)
65{
66 std::string string_type;
67
68 switch(enum_type)
69 {
70 // clang-format off
71 case OperatorType::CoupledNonlinearResidual: string_type = "CoupledNonlinearResidual"; break;
72 case OperatorType::CoupledLinearized: string_type = "CoupledLinearized"; break;
73 case OperatorType::PressurePoissonOperator: string_type = "PressurePoissonOperator"; break;
74 case OperatorType::ConvectiveOperator: string_type = "ConvectiveOperator"; break;
75 case OperatorType::HelmholtzOperator: string_type = "HelmholtzOperator"; break;
76 case OperatorType::ProjectionOperator: string_type = "ProjectionOperator"; break;
77 case OperatorType::VelocityConvDiffOperator: string_type = "VelocityConvDiffOperator"; break;
78 case OperatorType::InverseMassOperator: string_type = "InverseMassOperator"; break;
79
80 default:AssertThrow(false, dealii::ExcMessage("Not implemented.")); break;
81 // clang-format on
82 }
83
84 return string_type;
85}

87inline void
88string_to_enum(OperatorType & enum_type, std::string const string_type)
89{
90 // clang-format off
91 if (string_type == "CoupledNonlinearResidual") enum_type = OperatorType::CoupledNonlinearResidual;
92 else if(string_type == "CoupledLinearized") enum_type = OperatorType::CoupledLinearized;
93 else if(string_type == "PressurePoissonOperator") enum_type = OperatorType::PressurePoissonOperator;
94 else if(string_type == "ConvectiveOperator") enum_type = OperatorType::ConvectiveOperator;
95 else if(string_type == "HelmholtzOperator") enum_type = OperatorType::HelmholtzOperator;
96 else if(string_type == "ProjectionOperator") enum_type = OperatorType::ProjectionOperator;
97 else if(string_type == "VelocityConvDiffOperator") enum_type = OperatorType::VelocityConvDiffOperator;
98 else if(string_type == "InverseMassOperator") enum_type = OperatorType::InverseMassOperator;
99 else AssertThrow(false, dealii::ExcMessage("Unknown operator type. Not implemented."));
100 // clang-format on
101}

103inline unsigned int
104get_dofs_per_element(std::string const & input_file,
105 unsigned int const dim,
106 unsigned int const degree)
107{
108 std::string operator_type_string, pressure_degree = "MixedOrder";
109
110 dealii::ParameterHandler prm;
111 // clang-format off
112 prm.enter_subsection("Discretization");
113 prm.add_parameter("PressureDegree",
114 pressure_degree,
115 "Degree of pressure shape functions.",
116 dealii::Patterns::Selection("MixedOrder|EqualOrder"),
117 true);
118 prm.leave_subsection();
119 prm.enter_subsection("Throughput");
120 prm.add_parameter("OperatorType",
121 operator_type_string,
122 "Type of operator.",
123 dealii::Patterns::Anything(),
124 true);
125 prm.leave_subsection();
126 // clang-format on
127 prm.parse_input(input_file, "", true, true);
128
129 OperatorType operator_type;
130 string_to_enum(operator_type, operator_type_string);
131
132 unsigned int const velocity_dofs_per_element = dim * dealii::Utilities::pow(degree + 1, dim);
133 unsigned int pressure_dofs_per_element = 1;
134 if(pressure_degree == "MixedOrder")
135 pressure_dofs_per_element = dealii::Utilities::pow(degree, dim);
136 else if(pressure_degree == "EqualOrder")
137 pressure_dofs_per_element = dealii::Utilities::pow(degree + 1, dim);
138 else
139 AssertThrow(false, dealii::ExcMessage("Not implemented."));
140
141 if(operator_type == OperatorType::CoupledNonlinearResidual ||
142 operator_type == OperatorType::CoupledLinearized)
143 {
144 return velocity_dofs_per_element + pressure_dofs_per_element;
145 }
146 // velocity
147 else if(operator_type == OperatorType::ConvectiveOperator ||
148 operator_type == OperatorType::VelocityConvDiffOperator ||
149 operator_type == OperatorType::HelmholtzOperator ||
150 operator_type == OperatorType::ProjectionOperator ||
151 operator_type == OperatorType::InverseMassOperator)
152 {
153 return velocity_dofs_per_element;
154 }
155 // pressure
156 else if(operator_type == OperatorType::PressurePoissonOperator)
157 {
158 return pressure_dofs_per_element;
159 }
160 else
161 {
162 AssertThrow(false, dealii::ExcMessage("Not implemented."));
163 }
164
165 return 0;
166}
