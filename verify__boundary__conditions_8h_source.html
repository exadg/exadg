32 template<int dim, typename BoundaryDescriptor>
33 void
34 verify_boundary_conditions(BoundaryDescriptor const & boundary_descriptor, Grid<dim> const & grid)
35 {
36  // fill set with periodic boundary ids
37  std::set<types::boundary_id> periodic_boundary_ids;
38  for(auto periodic_pair : grid.periodic_faces)
39  {
40  AssertThrow(periodic_pair.cell[0]->level() == 0,
41  ExcMessage("Received periodic face pair on non-zero level"));
42 
43  periodic_boundary_ids.insert(
44  periodic_pair.cell[0]->face(periodic_pair.face_idx[0])->boundary_id());
45  periodic_boundary_ids.insert(
46  periodic_pair.cell[1]->face(periodic_pair.face_idx[1])->boundary_id());
47  }
48 
49  // Make sure that each boundary face has exactly one boundary type
50  for(auto cell : *grid.triangulation)
51  {
52  for(unsigned int f = 0; f < GeometryInfo<dim>::faces_per_cell; ++f)
53  {
54  if(cell.at_boundary(f))
55  {
56  types::boundary_id const boundary_id = cell.face(f)->boundary_id();
57  boundary_descriptor.verify_boundary_conditions(boundary_id, periodic_boundary_ids);
58  }
59  }
60  }
61 }
