17  * | | | | | /

20 template<typename VectorType>
21 void
22 interpolate(VectorType & dst,
23  double const & time,
24  std::vector<VectorType const *> const & solutions,
25  std::vector<double> const & times)
26 {
27  dst = 0;
28 
29  // loop over all interpolation points
30  for(unsigned int k = 0; k < solutions.size(); ++k)
31  {
32  // evaluate Lagrange polynomial l_k
33  double l_k = 1.0;
34 
35  for(unsigned int j = 0; j < solutions.size(); ++j)
36  {
37  if(j != k)
38  {
39  l_k *= (time - times[j]) / (times[k] - times[j]);
40  }
41  }
42 
43  dst.add(l_k, *solutions[k]);
44  }
45 }
