106template<int dim, typename Number>
107void
108SolverFluid<dim, Number>::setup(std::shared_ptr<FluidFSI::ApplicationBase<dim, Number>> application,
109 MPI_Comm const mpi_comm,
110 bool const is_test)
111{
112 // ALE: initialize PDE operator
113 if(application->get_parameters().mesh_movement_type == IncNS::MeshMovementType::Poisson)
114 {
115 ale_poisson_operator = std::make_shared<Poisson::Operator<dim, dim, Number>>(
116 application->get_grid(),
117 application->get_boundary_descriptor_ale_poisson(),
118 application->get_field_functions_ale_poisson(),
119 application->get_parameters_ale_poisson(),
120 "Poisson",
121 mpi_comm);
122 }
123 else if(application->get_parameters().mesh_movement_type == IncNS::MeshMovementType::Elasticity)
124 {
125 ale_elasticity_operator = std::make_shared<Structure::Operator<dim, Number>>(
126 application->get_grid(),
127 application->get_boundary_descriptor_ale_elasticity(),
128 application->get_field_functions_ale_elasticity(),
129 application->get_material_descriptor_ale_elasticity(),
130 application->get_parameters_ale_elasticity(),
131 "ale_elasticity",
132 mpi_comm);
133 }
134 else
135 {
136 AssertThrow(false, dealii::ExcMessage("not implemented."));
137 }
138
139 // ALE: initialize matrix_free_data
140 ale_matrix_free_data = std::make_shared<MatrixFreeData<dim, Number>>();
141
142 if(application->get_parameters().mesh_movement_type == IncNS::MeshMovementType::Poisson)
143 {
144 if(application->get_parameters_ale_poisson().enable_cell_based_face_loops)
145 Categorization::do_cell_based_loops(*application->get_grid()->triangulation,
146 ale_matrix_free_data->data);
147
148 ale_matrix_free_data->append(ale_poisson_operator);
149 }
150 else if(application->get_parameters().mesh_movement_type == IncNS::MeshMovementType::Elasticity)
151 {
152 ale_matrix_free_data->append(ale_elasticity_operator);
153 }
154 else
155 {
156 AssertThrow(false, dealii::ExcMessage("not implemented."));
157 }
158
159 // ALE: initialize matrix_free
160 ale_matrix_free = std::make_shared<dealii::MatrixFree<dim, Number>>();
161 ale_matrix_free->reinit(*application->get_grid()->mapping,
162 ale_matrix_free_data->get_dof_handler_vector(),
163 ale_matrix_free_data->get_constraint_vector(),
164 ale_matrix_free_data->get_quadrature_vector(),
165 ale_matrix_free_data->data);
166
167 // ALE: setup PDE operator and solver
168 if(application->get_parameters().mesh_movement_type == IncNS::MeshMovementType::Poisson)
169 {
170 ale_poisson_operator->setup(ale_matrix_free, ale_matrix_free_data);
171 ale_poisson_operator->setup_solver();
172 }
173 else if(application->get_parameters().mesh_movement_type == IncNS::MeshMovementType::Elasticity)
174 {
175 ale_elasticity_operator->setup(ale_matrix_free, ale_matrix_free_data);
176 ale_elasticity_operator->setup_solver();
177 }
178 else
179 {
180 AssertThrow(false, dealii::ExcMessage("not implemented."));
181 }
182
183 // ALE: create grid motion object
184 if(application->get_parameters().mesh_movement_type == IncNS::MeshMovementType::Poisson)
185 {
186 ale_grid_motion =
187 std::make_shared<GridMotionPoisson<dim, Number>>(application->get_grid()->mapping,
188 ale_poisson_operator);
189 }
190 else if(application->get_parameters().mesh_movement_type == IncNS::MeshMovementType::Elasticity)
191 {
192 ale_grid_motion = std::make_shared<GridMotionElasticity<dim, Number>>(
193 application->get_grid()->mapping,
194 ale_elasticity_operator,
195 application->get_parameters_ale_elasticity());
196 }
197 else
198 {
199 AssertThrow(false, dealii::ExcMessage("not implemented."));
200 }
201
202 // initialize pde_operator
203 pde_operator = IncNS::create_operator<dim, Number>(application->get_grid(),
204 ale_grid_motion,
205 application->get_boundary_descriptor(),
206 application->get_field_functions(),
207 application->get_parameters(),
208 "fluid",
209 mpi_comm);
210
211 // initialize matrix_free
212 matrix_free_data = std::make_shared<MatrixFreeData<dim, Number>>();
213 matrix_free_data->append(pde_operator);
214
215 matrix_free = std::make_shared<dealii::MatrixFree<dim, Number>>();
216 if(application->get_parameters().use_cell_based_face_loops)
217 Categorization::do_cell_based_loops(*application->get_grid()->triangulation,
218 matrix_free_data->data);
219 std::shared_ptr<dealii::Mapping<dim> const> mapping =
220 get_dynamic_mapping<dim, Number>(application->get_grid(), ale_grid_motion);
221 matrix_free->reinit(*mapping,
222 matrix_free_data->get_dof_handler_vector(),
223 matrix_free_data->get_constraint_vector(),
224 matrix_free_data->get_quadrature_vector(),
225 matrix_free_data->data);
226
227 // setup Navier-Stokes operator
228 pde_operator->setup(matrix_free, matrix_free_data);
229
230 // setup postprocessor
231 postprocessor = application->create_postprocessor();
232 postprocessor->setup(*pde_operator);
233
234 // setup time integrator before calling setup_solvers (this is necessary since the setup
235 // of the solvers depends on quantities such as the time_step_size or gamma0!!!)
236 AssertThrow(application->get_parameters().solver_type == IncNS::SolverType::Unsteady,
237 dealii::ExcMessage("Invalid parameter in context of fluid-structure interaction."));
238
239 // initialize time_integrator
240 time_integrator = IncNS::create_time_integrator<dim, Number>(
241 pde_operator, application->get_parameters(), mpi_comm, is_test, postprocessor);
242
243 time_integrator->setup(application->get_parameters().restarted_simulation);
244
245 pde_operator->setup_solvers(time_integrator->get_scaling_factor_time_derivative_term(),
246 time_integrator->get_velocity());
247}

249template<int dim, typename Number>
250void
251SolverFluid<dim, Number>::solve_ale(
252 std::shared_ptr<FluidFSI::ApplicationBase<dim, Number>> application,
253 bool const is_test) const
254{
255 dealii::Timer timer;
256 timer.restart();
257
258 dealii::Timer sub_timer;
259
260 sub_timer.restart();
261 bool const print_solver_info = time_integrator->print_solver_info();
262 ale_grid_motion->update(time_integrator->get_next_time(), print_solver_info and not(is_test));
263 timer_tree->insert({"ALE", "Solve and reinit mapping"}, sub_timer.wall_time());
264
265 sub_timer.restart();
266 std::shared_ptr<dealii::Mapping<dim> const> mapping =
267 get_dynamic_mapping<dim, Number>(application->get_grid(), ale_grid_motion);
268 matrix_free->update_mapping(*mapping);
269 timer_tree->insert({"ALE", "Update matrix-free"}, sub_timer.wall_time());
270
271 sub_timer.restart();
272 pde_operator->update_after_grid_motion();
273 timer_tree->insert({"ALE", "Update operator"}, sub_timer.wall_time());
274
275 sub_timer.restart();
276 time_integrator->ale_update();
277 timer_tree->insert({"ALE", "Update time integrator"}, sub_timer.wall_time());
278
279 timer_tree->insert({"ALE"}, timer.wall_time());
280}

282template<int dim, typename Number>
283std::shared_ptr<TimerTree>
284SolverFluid<dim, Number>::get_timings_ale() const
285{
286 return timer_tree;
287}
