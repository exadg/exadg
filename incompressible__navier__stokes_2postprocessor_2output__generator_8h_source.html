34/*
35 * Average velocity field over time for statistically steady, turbulent
36 * flow problems in order to visualize the time-averaged velocity field.
37 * Of course, this module can also be used for statistically unsteady problems,
38 * but in this case the mean velocity field is probably not meaningful.
39 */
40struct OutputDataMeanVelocity
41{
42 OutputDataMeanVelocity()
43 : calculate(false), sample_start_time(0.0), sample_end_time(0.0), sample_every_timesteps(1)
44 {
45 }
46
47 void
48 print(ConditionalOStream & pcout, bool unsteady)
49 {
50 if(unsteady)
51 {
52 print_parameter(pcout, "Calculate mean velocity", calculate);
53 print_parameter(pcout, " Sample start time", sample_start_time);
54 print_parameter(pcout, " Sample end time", sample_end_time);
55 print_parameter(pcout, " Sample every timesteps", sample_every_timesteps);
56 }
57 }
58
59 // calculate mean velocity field (for statistically steady, turbulent flows)
60 bool calculate;
61
62 // sampling information
63 double sample_start_time;
64 double sample_end_time;
65 unsigned int sample_every_timesteps;
66};

68struct OutputData : public OutputDataBase
69{
70 OutputData()
71 : write_vorticity(false),
72 write_divergence(false),
73 write_velocity_magnitude(false),
74 write_vorticity_magnitude(false),
75 write_streamfunction(false),
76 write_q_criterion(false),
77 mean_velocity(OutputDataMeanVelocity()),
78 write_cfl(false),
79 write_aspect_ratio(false)
80 {
81 }
82
83 void
84 print(ConditionalOStream & pcout, bool unsteady)
85 {
86 OutputDataBase::print(pcout, unsteady);
87
88 print_parameter(pcout, "Write vorticity", write_vorticity);
89 print_parameter(pcout, "Write divergence", write_divergence);
90 print_parameter(pcout, "Write velocity magnitude", write_velocity_magnitude);
91 print_parameter(pcout, "Write vorticity magnitude", write_vorticity_magnitude);
92 print_parameter(pcout, "Write streamfunction", write_streamfunction);
93 print_parameter(pcout, "Write Q criterion", write_q_criterion);
94
95 mean_velocity.print(pcout, unsteady);
96 }
97
98 // write vorticity of velocity field
99 bool write_vorticity;
100
101 // write divergence of velocity field
102 bool write_divergence;
103
104 // write velocity magnitude
105 bool write_velocity_magnitude;
106
107 // write vorticity magnitude
108 bool write_vorticity_magnitude;
109
110 // Calculate streamfunction in order to visualize streamlines!
111 // Note that this option is only available in 2D!
112 // To calculate the streamfunction Psi, a Poisson equation is solved
113 // with homogeneous Dirichlet BC's. Accordingly, this approach can only be
114 // used for flow problems where the whole boundary is one streamline, e.g.,
115 // cavity-type flow problems where the velocity is tangential to the boundary
116 // on one part of the boundary and 0 (no-slip) on the rest of the boundary.
117 bool write_streamfunction;
118
119 // write Q criterion
120 bool write_q_criterion;
121
122 // calculate mean velocity field (averaged over time)
123 OutputDataMeanVelocity mean_velocity;
124
125 // write cfl
126 bool write_cfl;
127
128 // write aspect ratio
129 bool write_aspect_ratio;
130};
