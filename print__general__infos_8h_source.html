96template<typename Number>
97inline void
98print_matrixfree_info(dealii::ConditionalOStream const & pcout)
99{
100 unsigned int const n_vect_doubles = dealii::VectorizedArray<Number>::size();
101 unsigned int const n_vect_bits = 8 * sizeof(Number) * n_vect_doubles;
102 std::string const vect_level = dealii::Utilities::System::get_current_vectorization_level();
103 std::string const type = get_type(Number());
104
105 // clang-format off
106 pcout << std::endl
107 << "MatrixFree info:" << std::endl
108 << std::endl
109 << " vectorization level = " << DEAL_II_COMPILER_VECTORIZATION_LEVEL
110 << std::endl
111 << " Vectorization over "
112 << n_vect_doubles << " " << type << " = " << n_vect_bits << " bits (" << vect_level << ")"
113 << std::endl;
114 // clang-format on
115}

117template<int dim>
118inline void
119print_grid_info(dealii::ConditionalOStream const & pcout, Grid<dim> const & grid)
120{
121 pcout << std::endl
122 << "Generating grid for " << dim << "-dimensional problem:" << std::endl
123 << std::endl;
124
125 print_parameter(pcout, "Max. number of refinements", grid.triangulation->n_global_levels() - 1);
126 print_parameter(pcout, "Number of cells", grid.triangulation->n_global_active_cells());
127
128 std::shared_ptr<dealii::MappingQGeneric<dim>> mapping_q_generic =
129 std::dynamic_pointer_cast<dealii::MappingQGeneric<dim>>(grid.mapping);
130 if(mapping_q_generic.get() != 0)
131 print_parameter(pcout, "Mapping degree", mapping_q_generic->get_degree());
132}

134template<typename Number>
135inline void
136print_general_info(dealii::ConditionalOStream const & pcout,
137 MPI_Comm const & mpi_comm,
138 bool const is_test)
139{
140 print_exadg_header(pcout);
141
142 if(not(is_test))
143 {
144 print_dealii_info(pcout);
145 print_matrixfree_info<Number>(pcout);
146 }
147
148 print_MPI_info(pcout, mpi_comm);
149}
