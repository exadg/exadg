130  /*
131  * Smagorinsky model (1963):
132  *
133  * nu_SGS = (C * filter_width)^{2} * sqrt(2 * S:S)
134  *
135  * where S is the symmetric part of the velocity gradient
136  *
137  * S = 1/2 * (grad(u) + grad(u)^T) and S:S = S_ij * S_ij
138  *
139  * and the model constant is
140  *
141  * C = 0.165 (Nicoud et al. (2011))
142  * C = 0.18 (Toda et al. (2010))
143  */
144  void
145  smagorinsky_model(scalar const & filter_width,
146  tensor const & velocity_gradient,
147  double const & C,
148  scalar & viscosity) const;
149 
150  /*
151  * Vreman model (2004): Note that we only consider the isotropic variant of the Vreman model:
152  *
153  * nu_SGS = (C * filter_width)^{2} * D
154  *
155  * where the differential operator D is defined as
156  *
157  * D = sqrt(B_gamma / ||grad(u)||^{2})
158  *
159  * with
160  *
161  * ||grad(u)||^{2} = grad(u) : grad(u) and grad(u) = d(u_i)/d(x_j) ,
162  *
163  * gamma = grad(u) * grad(u)^T ,
164  *
165  * and
166  *
167  * B_gamma = gamma_11 * gamma_22 - gamma_12^{2}
168  * +gamma_11 * gamma_33 - gamma_13^{2}
169  * +gamma_22 * gamma_33 - gamma_23^{2}
170  *
171  * Note that if ||grad(u)||^{2} = 0, nu_SGS is consistently defined as zero.
172  *
173  */
174  void
175  vreman_model(scalar const & filter_width,
176  tensor const & velocity_gradient,
177  double const & C,
178  scalar & viscosity) const;
179 
180  /*
181  * WALE (wall-adapting local eddy-viscosity) model (Nicoud & Ducros 1999):
182  *
183  * nu_SGS = (C * filter_width)^{2} * D ,
184  *
185  * where the differential operator D is defined as
186  *
187  * D = (S^{d}:S^{d})^{3/2} / ( (S:S)^{5/2} + (S^{d}:S^{d})^{5/4} )
188  *
189  * where S is the symmetric part of the velocity gradient
190  *
191  * S = 1/2 * (grad(u) + grad(u)^T) and S:S = S_ij * S_ij
192  *
193  * and S^{d} the traceless symmetric part of the square of the velocity
194  * gradient tensor
195  *
196  * S^{d} = 1/2 * (g^{2} + (g^{2})^T) - 1/3 * trace(g^{2}) * I
197  *
198  * with the square of the velocity gradient tensor
199  *
200  * g^{2} = grad(u) * grad(u)
201  *
202  * and the identity tensor I.
203  *
204  */
205  void
206  wale_model(scalar const & filter_width,
207  tensor const & velocity_gradient,
208  double const & C,
209  scalar & viscosity) const;
210 
211  /*
212  * Sigma model (Toda et al. 2010, Nicoud et al. 2011):
213  *
214  * nu_SGS = (C * filter_width)^{2} * D
215  *
216  * where the differential operator D is defined as
217  *
218  * D = s3 * (s1 - s2) * (s2 - s3) / s1^{2}
219  *
220  * where s1 >= s2 >= s3 >= 0 are the singular values of
221  * the velocity gradient tensor g = grad(u).
222  *
223  * The model constant is
224  *
225  * C = 1.35 (Nicoud et al. (2011)) ,
226  * C = 1.5 (Toda et al. (2010)) .
227  */
228  void
229  sigma_model(scalar const & filter_width,
230  tensor const & velocity_gradient,
231  double const & C,
232  scalar & viscosity) const;
