33namespace ExaDG

35namespace Structure
36{
37enum class BoundaryType
38{
39 Undefined,
40 Dirichlet,
41 DirichletCached,
42 Neumann,
43 NeumannCached
44};
45
46template<int dim>
47struct BoundaryDescriptor
48{
49 std::map<dealii::types::boundary_id, std::shared_ptr<dealii::Function<dim>>> dirichlet_bc;
50 std::map<dealii::types::boundary_id, dealii::ComponentMask> dirichlet_bc_component_mask;
51
52 // another type of Dirichlet boundary condition where the Dirichlet values come
53 // from the solution on another domain that is in contact with the actual domain
54 // of interest at the given boundary (this type of Dirichlet boundary condition
55 // is required for fluid-structure interaction problems)
56 std::map<dealii::types::boundary_id, std::shared_ptr<FunctionCached<1, dim>>> dirichlet_cached_bc;
57
58 std::map<dealii::types::boundary_id, std::shared_ptr<dealii::Function<dim>>> neumann_bc;
59
60 // another type of Neumann boundary condition where the traction force comes
61 // from the solution on another domain that is in contact with the actual domain
62 // of interest at the given boundary (this type of Neumann boundary condition
63 // is required for fluid-structure interaction problems)
64 std::map<dealii::types::boundary_id, std::shared_ptr<FunctionCached<1, dim>>> neumann_cached_bc;
65
66 inline DEAL_II_ALWAYS_INLINE //
67 BoundaryType
68 get_boundary_type(dealii::types::boundary_id const & boundary_id) const
69 {
70 if(this->dirichlet_bc.find(boundary_id) != this->dirichlet_bc.end())
71 return BoundaryType::Dirichlet;
72 else if(this->dirichlet_cached_bc.find(boundary_id) != this->dirichlet_cached_bc.end())
73 return BoundaryType::DirichletCached;
74 else if(this->neumann_bc.find(boundary_id) != this->neumann_bc.end())
75 return BoundaryType::Neumann;
76 else if(this->neumann_cached_bc.find(boundary_id) != this->neumann_cached_bc.end())
77 return BoundaryType::NeumannCached;
78
79 AssertThrow(false,
80 dealii::ExcMessage(
81 "Could not find a boundary type to the specified boundary_id = " +
82 std::to_string(boundary_id) +
83 ". A possible reason is that you "
84 "forgot to define a boundary condition for this boundary_id, or "
85 "that the boundary type associated to this boundary has not been "
86 "implemented."));
87
88 return BoundaryType::Undefined;
89 }
90
91 inline DEAL_II_ALWAYS_INLINE //
92 void
93 verify_boundary_conditions(
94 dealii::types::boundary_id const boundary_id,
95 std::set<dealii::types::boundary_id> const & periodic_boundary_ids) const
96 {
97 unsigned int counter = 0;
98 if(dirichlet_bc.find(boundary_id) != dirichlet_bc.end())
99 {
100 counter++;
101
102 AssertThrow(
103 dirichlet_bc_component_mask.find(boundary_id) != dirichlet_bc_component_mask.end(),
104 dealii::ExcMessage(
105 "dirichlet_bc_component_mask must contain the same boundary IDs as dirichlet_bc."));
106 }
107
108 if(dirichlet_cached_bc.find(boundary_id) != dirichlet_cached_bc.end())
109 counter++;
110
111 if(neumann_bc.find(boundary_id) != neumann_bc.end())
112 counter++;
113
114 if(neumann_cached_bc.find(boundary_id) != neumann_cached_bc.end())
115 counter++;
116
117 if(periodic_boundary_ids.find(boundary_id) != periodic_boundary_ids.end())
118 counter++;
119
120 AssertThrow(counter == 1,
121 dealii::ExcMessage("Boundary face with non-unique boundary type found."));
122 }
123};
