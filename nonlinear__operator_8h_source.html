83  /*
84  * Calculates the integral
85  *
86  * (v_h, factor * d_h)_Omega + (Grad(v_h), P_h)_Omega
87  *
88  * with 1st Piola-Kirchhoff stress tensor P_h
89  *
90  * P_h = F_h * S_h ,
91  *
92  * 2nd Piola-Kirchhoff stress tensor S_h
93  *
94  * S_h = function(E_h) ,
95  *
96  * Green-Lagrange strain tensor E_h
97  *
98  * E_h = 1/2 (F_h^T * F_h - 1) ,
99  *
100  * material deformation gradient F_h
101  *
102  * F_h = 1 + Grad(d_h) ,
103  *
104  * where
105  *
106  * d_h denotes the displacement vector.
107  */
108  void
109  do_cell_integral_nonlinear(IntegratorCell & integrator) const;

134  /*
135  * Calculates the integral
136  *
137  * (v_h, factor * delta d_h)_Omega + (Grad(v_h), delta P_h)_Omega
138  *
139  * with the directional derivative of the 1st Piola-Kirchhoff stress tensor P_h
140  *
141  * delta P_h = d(P)/d(d)|_{d_lin} * delta d_h ,
142  *
143  * with the point of linearization
144  *
145  * d_lin ,
146  *
147  * and displacement increment
148  *
149  * delta d_h .
150  *
151  * Computing the linearization yields
152  *
153  * delta P_h = + Grad(delta_d) * S(d_lin)
154  * + F(d_lin) * (C_lin : (F^T(d_lin) * Grad(delta d))) .
155  *
156  * Note that a dependency of the Neumann BC on the displacements d through
157  * the area ratio da/dA = function(d) is neglected in the linearization.
158  */
