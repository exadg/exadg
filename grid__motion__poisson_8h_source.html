33namespace ExaDG
34{
38template<int dim, typename Number>
39class GridMotionPoisson : public GridMotionBase<dim, Number>
40{
41public:
42 typedef dealii::LinearAlgebra::distributed::Vector<Number> VectorType;
43
47 GridMotionPoisson(std::shared_ptr<dealii::Mapping<dim> const> mapping_undeformed,
48 std::shared_ptr<Poisson::Operator<dim, Number, dim>> poisson_operator)
49 : GridMotionBase<dim, Number>(mapping_undeformed,
50 // extract mapping_degree_moving from Poisson operator
51 poisson_operator->get_dof_handler().get_fe().degree,
52 poisson_operator->get_dof_handler().get_triangulation()),
53 poisson(poisson_operator),
54 pcout(std::cout,
55 dealii::Utilities::MPI::this_mpi_process(
56 poisson_operator->get_dof_handler().get_communicator()) == 0),
57 iterations({0, 0})
58 {
59 poisson->initialize_dof_vector(displacement);
60 }
61
65 void
66 update(double const time, bool const print_solver_info) override
67 {
68 dealii::Timer timer;
69 timer.restart();
70
71 VectorType rhs;
72 poisson->initialize_dof_vector(rhs);
73
74 // compute rhs and solve mesh deformation problem
75 poisson->rhs(rhs, time);
76
77 auto const n_iter = poisson->solve(displacement, rhs, time);
78 iterations.first += 1;
79 iterations.second += n_iter;
80
81 if(print_solver_info)
82 {
83 this->pcout << std::endl << "Solve moving mesh problem (Poisson):";
84 print_solver_info_linear(pcout, n_iter, timer.wall_time());
85 }
86
87 this->moving_mapping->initialize_mapping_q_cache(this->mapping_undeformed,
88 displacement,
89 poisson->get_dof_handler());
90 }
91
95 void
96 print_iterations() const override
97 {
98 std::vector<std::string> names;
99 std::vector<double> iterations_avg;
100
101 names = {"Linear iterations"};
102 iterations_avg.resize(1);
103 iterations_avg[0] = (double)iterations.second / std::max(1., (double)iterations.first);
104
105 print_list_of_iterations(pcout, names, iterations_avg);
106 }
107
108private:
109 std::shared_ptr<Poisson::Operator<dim, Number, dim>> poisson;
110
111 // store solution of previous time step / iteration so that a good initial
112 // guess is available in the next step, easing convergence or reducing computational
113 // costs by allowing larger tolerances
114 VectorType displacement;
115
116 dealii::ConditionalOStream pcout;
117
118 std::pair<unsigned int /* calls */, unsigned long long /* iteration counts */> iterations;
119};
