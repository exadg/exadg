29/*
30 * Class that provides a spherical manifold applied to one of the faces
31 * of a quadrilateral element.
32 * On the face subject to the spherical manifold intermediate points are
33 * inserted so that an equidistant distribution of points in terms of
34 * arclength is obtained.
35 * When refining the mesh, all child cells are subject to this "one-sided"
36 * spherical volume manifold.
37 * This manifold description is available for the two-dimensional case,
38 * and for the three-dimensional case with the restriction that the geometry
39 * has to be extruded in x3/z-direction.
40 */
41template<int dim>
42class OneSidedCylindricalManifold : public dealii::ChartManifold<dim, dim, dim>
43{
44public:
45 OneSidedCylindricalManifold(typename dealii::Triangulation<dim>::cell_iterator const & cell_in,
46 unsigned int const face_in,
47 dealii::Point<dim> const & center_in)
48 : alpha(1.0), radius(1.0), cell(cell_in), face(face_in), center(center_in)
49 {
50 AssertThrow(face >= 0 && face <= 3,
51 dealii::ExcMessage(
52 "One sided spherical manifold can only be applied to face f=0,1,2,3."));
53
54 // get center coordinates in x1-x2 plane
55 x_C[0] = center[0];
56 x_C[1] = center[1];
57
58 // determine x_1 and x_2 which denote the end points of the face that is
59 // subject to the spherical manifold.
60 dealii::Point<dim> x_1, x_2;
61 x_1 = cell->vertex(get_vertex_id(0));
62 x_2 = cell->vertex(get_vertex_id(1));
63
64 dealii::Point<2> x_1_2d = dealii::Point<2>(x_1[0], x_1[1]);
65 dealii::Point<2> x_2_2d = dealii::Point<2>(x_2[0], x_2[1]);
66
67 initialize(x_1_2d, x_2_2d);
68 }
69
70 void initialize(dealii::Point<2> const & x_1, dealii::Point<2> const & x_2)
71 {
72 double const tol = 1.e-12;
73
74 v_1 = x_1 - x_C;
75 v_2 = x_2 - x_C,
76
77 // calculate radius of spherical manifold
78 radius = v_1.norm();
79
80 // check correctness of geometry and parameters
81 double radius_check = v_2.norm();
82 AssertThrow(std::abs(radius - radius_check) < tol * radius,
83 dealii::ExcMessage(
84 "Invalid geometry parameters. To apply a spherical manifold both "
85 "end points of the face must have the same distance from the center."));
86
87 // normalize v_1 and v_2
88 v_1 /= v_1.norm();
89 v_2 /= v_2.norm();
90
91 // calculate angle between v_1 and v_2
92 alpha = std::acos(v_1 * v_2);
93
94 // calculate vector that is perpendicular to v_1 in plane that is spanned by v_1 and v_2
95 normal = v_2 - (v_2 * v_1) * v_1;
96
97 AssertThrow(normal.norm() > tol, dealii::ExcMessage("Vector must not have length 0."));
98
99 normal /= normal.norm();
100 }

102 /*
103 * push_forward operation that maps point xi in reference coordinates [0,1]^d to
104 * point x in physical coordinates
105 */
106 dealii::Point<dim>
107 push_forward(dealii::Point<dim> const & xi) const override
108 {
109 dealii::Point<dim> x;
110
111 // standard mapping from reference space to physical space using d-linear shape functions
112 for(unsigned int const v : cell->vertex_indices())
113 {
114 double shape_function_value = cell->reference_cell().d_linear_shape_function(xi, v);
115 x += shape_function_value * cell->vertex(v);
116 }
117
118 // Add contribution of spherical manifold.
119 // Here, we only operate in the xi1-xi2 plane.
120
121 // set xi_face, xi_other to xi[0],xi[1] depending on the face that is subject to the manifold
122 unsigned int index_face = get_index_face();
123 unsigned int index_other = get_index_other();
124 double const xi_face = xi[index_face];
125 double const xi_other = xi[index_other];
126
127 // calculate deformation related to the spherical manifold
128 double beta = xi_face * alpha;
129
130 dealii::Tensor<1, 2> direction;
131 direction = std::cos(beta) * v_1 + std::sin(beta) * normal;
132
133 Assert(std::abs(direction.norm() - 1.0) < 1.e-12,
134 dealii::ExcMessage("Vector must have length 1."));
135
136 // calculate point x_S on spherical manifold
137 dealii::Tensor<1, 2> x_S;
138 x_S = x_C + radius * direction;
139
140 // calculate displacement as compared to straight sided quadrilateral element
141 // on the face that is subject to the manifold
142 dealii::Tensor<1, 2> displ, x_lin;
143 for(unsigned int v : dealii::ReferenceCells::template get_hypercube<1>().vertex_indices())
144 {
145 double shape_function_value =
146 dealii::ReferenceCells::template get_hypercube<1>().d_linear_shape_function(
147 dealii::Point<1>(xi_face), v);
148
149 unsigned int vertex_id = get_vertex_id(v);
150 dealii::Point<dim> vertex = cell->vertex(vertex_id);
151
152 x_lin[0] += shape_function_value * vertex[0];
153 x_lin[1] += shape_function_value * vertex[1];
154 }
155
156 displ = x_S - x_lin;
157
158 // deformation decreases linearly in the second (other) direction
159 dealii::Point<1> xi_other_1d = dealii::Point<1>(xi_other);
160 unsigned int index_1d = get_index_1d();
161 double fading_value =
162 dealii::ReferenceCells::template get_hypercube<1>().d_linear_shape_function(xi_other_1d,
163 index_1d);
164 x[0] += fading_value * displ[0];
165 x[1] += fading_value * displ[1];
166
167 Assert(dealii::numbers::is_finite(x.norm_square()), dealii::ExcMessage("Invalid point found"));
168
169 return x;
170 }
171
172 /*
173 * Calculate vertex_id of 2d object (cell in 2d, face4 in 3d)
174 * given the vertex_id of the 1d object (vertex_id_1d = 0,1).
175 */
176 unsigned int
177 get_vertex_id(unsigned int vertex_id_1d) const
178 {
179 unsigned int vertex_id = 0;
180
181 if(face == 0)
182 vertex_id = 2 * vertex_id_1d;
183 else if(face == 1)
184 vertex_id = 1 + 2 * vertex_id_1d;
185 else if(face == 2)
186 vertex_id = vertex_id_1d;
187 else if(face == 3)
188 vertex_id = 2 + vertex_id_1d;
189
190 return vertex_id;
191 }
192
193 /*
194 * Calculate index of 1d linear shape function (0 or 1)
195 * that takes a value of 1 on the specified face.
196 */
197 unsigned int
198 get_index_1d() const
199 {
200 unsigned int index_1d = 0;
201
202 if(face == 0 || face == 2)
203 index_1d = 0;
204 else if(face == 1 || face == 3)
205 index_1d = 1;
206 else
207 Assert(false, dealii::ExcMessage("Face ID is invalid."));
208
209 return index_1d;
210 }
211
212 /*
213 * Calculate which xi-coordinate corresponds to the
214 * tangent direction of the respective face
215 */
216 unsigned int
217 get_index_face() const
218 {
219 unsigned int index_face = 0;
220
221 if(face == 0 || face == 1)
222 index_face = 1;
223 else if(face == 2 || face == 3)
224 index_face = 0;
225 else
226 Assert(false, dealii::ExcMessage("Face ID is invalid."));
227
228 return index_face;
229 }
230
231 /*
232 * Calculate which xi-coordinate corresponds to
233 * the normal direction of the respective face
234 * in xi1-xi2-plane.
235 */
236 unsigned int
237 get_index_other() const
238 {
239 return 1 - get_index_face();
240 }

242 /*
243 * This function calculates the inverse Jacobi matrix dx/d(xi) = d(phi(xi))/d(xi) of
244 * the push-forward operation phi: [0,1]^d -> R^d: xi -> x = phi(xi)
245 * We assume that the gradient of the standard bilinear shape functions is sufficient
246 * to find the solution.
247 */
248 dealii::Tensor<2, dim>
249 get_inverse_jacobian(dealii::Point<dim> const & xi) const
250 {
251 dealii::Tensor<2, dim> jacobian;
252
253 // standard mapping from reference space to physical space using d-linear shape functions
254 for(unsigned int const v : cell->vertex_indices())
255 {
256 dealii::Tensor<1, dim> shape_function_gradient =
257 cell->reference_cell().d_linear_shape_function_gradient(xi, v);
258 jacobian += outer_product(cell->vertex(v), shape_function_gradient);
259 }
260
261 return invert(jacobian);
262 }

264 /*
265 * pull_back operation that maps point x in physical coordinates
266 * to point xi in reference coordinates [0,1]^d using the
267 * push_forward operation and Newton's method
268 */
269 dealii::Point<dim>
270 pull_back(dealii::Point<dim> const & x) const override
271 {
272 dealii::Point<dim> xi;
273 dealii::Tensor<1, dim> residual = push_forward(xi) - x;
274 dealii::Tensor<1, dim> delta_xi;
275
276 // Newton method to solve nonlinear pull_back operation
277 unsigned int n_iter = 0, MAX_ITER = 100;
278 double const TOL = 1.e-12;
279 while(residual.norm() > TOL && n_iter < MAX_ITER)
280 {
281 // multiply by -1.0, i.e., shift residual to the rhs
282 residual *= -1.0;
283
284 // solve linear problem
285 delta_xi = get_inverse_jacobian(xi) * residual;
286
287 // add increment
288 xi += delta_xi;
289
290 // make sure that xi is in the valid range [0,1]^d
291 if(xi[0] < 0.0)
292 xi[0] = 0.0;
293 else if(xi[0] > 1.0)
294 xi[0] = 1.0;
295
296 if(xi[1] < 0.0)
297 xi[1] = 0.0;
298 else if(xi[1] > 1.0)
299 xi[1] = 1.0;
300
301 // evaluate residual
302 residual = push_forward(xi) - x;
303
304 // increment counter
305 ++n_iter;
306 }
307
308 Assert(n_iter < MAX_ITER,
309 dealii::ExcMessage("Newton solver did not converge to given tolerance. "
310 "Maximum number of iterations exceeded."));
311
312 Assert(xi[0] >= 0.0 && xi[0] <= 1.0,
313 dealii::ExcMessage("Pull back operation generated invalid xi[0] values."));
314
315 Assert(xi[1] >= 0.0 && xi[1] <= 1.0,
316 dealii::ExcMessage("Pull back operation generated invalid xi[1] values."));
317
318 return xi;
319 }
320
321 std::unique_ptr<dealii::Manifold<dim>>
322 clone() const override
323 {
324 return std::make_unique<OneSidedCylindricalManifold<dim>>(cell, face, center);
325 }
326
327private:
328 dealii::Point<2> x_C;
329 dealii::Tensor<1, 2> v_1;
330 dealii::Tensor<1, 2> v_2;
331 dealii::Tensor<1, 2> normal;
332 double alpha;
333 double radius;
334
335 typename dealii::Triangulation<dim>::cell_iterator cell;
336 unsigned int face;
337 dealii::Point<dim> center;
338};

418 /*
419 * push_forward operation that maps point xi in reference coordinates [0,1]^d to
420 * point x in physical coordinates
421 */
422 dealii::Point<dim>
423 push_forward(dealii::Point<dim> const & xi) const override
424 {
425 dealii::Point<dim> x;
426
427 // standard mapping from reference space to physical space using d-linear shape functions
428 for(unsigned int const v : cell->vertex_indices())
429 {
430 double shape_function_value = cell->reference_cell().d_linear_shape_function(xi, v);
431 x += shape_function_value * cell->vertex(v);
432 }
433
434 // Add contribution of conical manifold.
435 // Here, we only operate in the xi1-xi2 plane.
436
437 // set xi_face, xi_other to xi[0],xi[1] depending on the face that is subject to the manifold
438 unsigned int index_face = get_index_face();
439 unsigned int index_other = get_index_other();
440 double const xi_face = xi[index_face];
441 double const xi_other = xi[index_other];
442
443 // calculate deformation related to the conical manifold
444 double beta = xi_face * alpha;
445
446 dealii::Tensor<1, 2> direction;
447 direction = std::cos(beta) * v_1 + std::sin(beta) * normal;
448
449 Assert(std::abs(direction.norm() - 1.0) < 1.e-12,
450 dealii::ExcMessage("Vector must have length 1."));
451
452 // calculate point x_S on spherical manifold
453 dealii::Tensor<1, 2> x_S;
454 x_S = x_C + r_0 * direction;
455
456 // calculate displacement as compared to straight sided quadrilateral element
457 // on the face that is subject to the manifold
458 dealii::Tensor<1, 2> displ, x_lin;
459 for(unsigned int const v : dealii::ReferenceCells::template get_hypercube<1>().vertex_indices())
460 {
461 double shape_function_value =
462 dealii::ReferenceCells::template get_hypercube<1>().d_linear_shape_function(
463 dealii::Point<1>(xi_face), v);
464
465 unsigned int vertex_id = get_vertex_id(v);
466 dealii::Point<dim> vertex = cell->vertex(vertex_id);
467
468 x_lin[0] += shape_function_value * vertex[0];
469 x_lin[1] += shape_function_value * vertex[1];
470 }
471
472 displ = x_S - x_lin;
473
474 // conical manifold
475 displ *= (1 - xi[2] * (r_0 - r_1) / r_0);
476
477 // deformation decreases linearly in the second (other) direction
478 dealii::Point<1> xi_other_1d = dealii::Point<1>(xi_other);
479 unsigned int index_1d = get_index_1d();
480 double fading_value =
481 dealii::ReferenceCells::template get_hypercube<1>().d_linear_shape_function(xi_other_1d,
482 index_1d);
483 x[0] += fading_value * displ[0];
484 x[1] += fading_value * displ[1];
485
486 Assert(dealii::numbers::is_finite(x.norm_square()), dealii::ExcMessage("Invalid point found"));
487
488 return x;
489 }

561 /*
562 * This function calculates the inverse Jacobi matrix dx/d(xi) = d(phi(xi))/d(xi) of
563 * the push-forward operation phi: [0,1]^d -> R^d: xi -> x = phi(xi)
564 * We assume that the gradient of the standard bilinear shape functions is sufficient
565 * to find the solution.
566 */
567 dealii::Tensor<2, dim>
568 get_inverse_jacobian(dealii::Point<dim> const & xi) const
569 {
570 dealii::Tensor<2, dim> jacobian;
571
572 // standard mapping from reference space to physical space using d-linear shape functions
573 for(unsigned int const v : cell->vertex_indices())
574 {
575 dealii::Tensor<1, dim> shape_function_gradient =
576 cell->reference_cell().d_linear_shape_function_gradient(xi, v);

583 /*
584 * pull_back operation that maps point x in physical coordinates
585 * to point xi in reference coordinates [0,1]^d using the
586 * push_forward operation and Newton's method
587 */
588 dealii::Point<dim>
589 pull_back(dealii::Point<dim> const & x) const override
590 {
591 dealii::Point<dim> xi;
592 dealii::Tensor<1, dim> residual = push_forward(xi) - x;
593 dealii::Tensor<1, dim> delta_xi;
594
595 // Newton method to solve nonlinear pull_back operation
596 unsigned int n_iter = 0, MAX_ITER = 100;
597 double const TOL = 1.e-12;
598 while(residual.norm() > TOL && n_iter < MAX_ITER)
599 {
600 // multiply by -1.0, i.e., shift residual to the rhs
601 residual *= -1.0;
602
603 // solve linear problem
604 delta_xi = get_inverse_jacobian(xi) * residual;
605
606 // add increment
607 xi += delta_xi;
608
609 // make sure that xi is in the valid range [0,1]^d
610 if(xi[0] < 0.0)
611 xi[0] = 0.0;
612 else if(xi[0] > 1.0)
613 xi[0] = 1.0;
614
615 if(xi[1] < 0.0)
616 xi[1] = 0.0;
617 else if(xi[1] > 1.0)
618 xi[1] = 1.0;
619
620 if(xi[2] < 0.0)
621 xi[2] = 0.0;
622 else if(xi[2] > 1.0)
623 xi[2] = 1.0;
624
625 // evaluate residual
626 residual = push_forward(xi) - x;
627
628 // increment counter
629 ++n_iter;
630 }
631
632 Assert(n_iter < MAX_ITER,
633 dealii::ExcMessage("Newton solver did not converge to given tolerance. "
634 "Maximum number of iterations exceeded."));
635
636 Assert(xi[0] >= 0.0 && xi[0] <= 1.0,
637 dealii::ExcMessage("Pull back operation generated invalid xi[0] values."));
638
639 Assert(xi[1] >= 0.0 && xi[1] <= 1.0,
640 dealii::ExcMessage("Pull back operation generated invalid xi[1] values."));
641
642 Assert(xi[2] >= 0.0 && xi[2] <= 1.0,
643 dealii::ExcMessage("Pull back operation generated invalid xi[2] values."));
644
645 return xi;
646 }

672/*
673 * Own implementation of cylindrical manifold
674 * with an equidistant distribution of nodes along the
675 * cylinder surface.
676 */
677template<int dim, int spacedim = dim>
678class MyCylindricalManifold : public dealii::ChartManifold<dim, spacedim, spacedim>
679{
680public:
681 MyCylindricalManifold(dealii::Point<spacedim> const center_in)
682 : dealii::ChartManifold<dim, spacedim, spacedim>(
683 MyCylindricalManifold<dim, spacedim>::get_periodicity()),
684 center(center_in)
685 {
686 }
687
688 dealii::Tensor<1, spacedim>
689 get_periodicity()
690 {
691 dealii::Tensor<1, spacedim> periodicity;
692
693 // angle theta is 2*pi periodic
694 periodicity[1] = 2 * dealii::numbers::PI;
695 return periodicity;
696 }
697
698 dealii::Point<spacedim>
699 push_forward(dealii::Point<spacedim> const & ref_point) const override
700 {
701 double const radius = ref_point[0];
702 double const theta = ref_point[1];
703
704 Assert(ref_point[0] >= 0.0, dealii::ExcMessage("Radius must be positive."));
705
706 dealii::Point<spacedim> space_point;
707 if(radius > 1e-10)
708 {
709 AssertThrow(spacedim == 2 || spacedim == 3,
710 dealii::ExcMessage("Only implemented for 2D and 3D case."));
711
712 space_point[0] = radius * cos(theta);
713 space_point[1] = radius * sin(theta);
714
715 if(spacedim == 3)
716 space_point[2] = ref_point[2];
717 }
718
719 return space_point + center;
720 }
721
722 dealii::Point<spacedim>
723 pull_back(dealii::Point<spacedim> const & space_point) const override
724 {
725 dealii::Tensor<1, spacedim> vector;
726 vector[0] = space_point[0] - center[0];
727 vector[1] = space_point[1] - center[1];
728 // for the 3D case: vector[2] will always be 0.
729
730 double const radius = vector.norm();
731
732 dealii::Point<spacedim> ref_point;
733 ref_point[0] = radius;
734 ref_point[1] = atan2(vector[1], vector[0]);
735 if(ref_point[1] < 0)
736 ref_point[1] += 2.0 * dealii::numbers::PI;
737 if(spacedim == 3)
738 ref_point[2] = space_point[2];
739
740 return ref_point;
741 }
