40template<int dim, int spacedim>
41std::unique_ptr<MPI_Comm, void (*)(MPI_Comm *)>
42create_subcommunicator(DoFHandler<dim, spacedim> const & dof_handler)
43{
44 unsigned int n_locally_owned_cells = 0;
45 for(const auto & cell : dof_handler.active_cell_iterators())
46 if(cell->is_locally_owned())
47 ++n_locally_owned_cells;
48
49 MPI_Comm const mpi_comm = dof_handler.get_communicator();
50
51 // In case some of the MPI ranks do not have cells, we create a
52 // sub-communicator to exclude all those processes from the MPI
53 // communication in the matrix-based operation sand hence speed up those
54 // operations. Note that we have to free the communicator again, which is
55 // done by a custom deleter of the unique pointer that is run when it goes
56 // out of scope.
57 if(Utilities::MPI::min(n_locally_owned_cells, mpi_comm) == 0)
58 {
59 std::unique_ptr<MPI_Comm, void (*)(MPI_Comm *)> subcommunicator(new MPI_Comm,
60 [](MPI_Comm * comm) {
61 MPI_Comm_free(comm);
62 delete comm;
63 });
64 MPI_Comm_split(mpi_comm,
65 n_locally_owned_cells > 0,
66 Utilities::MPI::this_mpi_process(mpi_comm),
67 subcommunicator.get());
68
69 return subcommunicator;
70 }
71 else
72 {
73 std::unique_ptr<MPI_Comm, void (*)(MPI_Comm *)> communicator(new MPI_Comm, [](MPI_Comm * comm) {
74 delete comm;
75 });
76 *communicator = mpi_comm;
77
78 return communicator;
79 }
80}

83template<typename Operator, typename Number>
84class PreconditionerML : public PreconditionerBase<Number>
85{
86private:
87 typedef LinearAlgebra::distributed::Vector<Number> VectorType;
88
89 typedef TrilinosWrappers::PreconditionAMG::AdditionalData MLData;
90
91public:
92 // distributed sparse system matrix
93 TrilinosWrappers::SparseMatrix system_matrix;
94
95private:
96 TrilinosWrappers::PreconditionAMG amg;
97
98public:
99 PreconditionerML(Operator const & op, MLData ml_data = MLData())
100 : pde_operator(op), ml_data(ml_data)
101 {
102 // initialize system matrix
103 pde_operator.init_system_matrix(system_matrix,
104 op.get_matrix_free().get_dof_handler().get_communicator());
105
106 // calculate_matrix
107 pde_operator.calculate_system_matrix(system_matrix);
108
109 // initialize Trilinos' AMG
110 amg.initialize(system_matrix, ml_data);
111 }
112
113 TrilinosWrappers::SparseMatrix const &
114 get_system_matrix()
115 {
116 return system_matrix;
117 }
118
119 void
120 update() override
121 {
122 // clear content of matrix since the next calculate_system_matrix-commands add their result
123 system_matrix *= 0.0;
124
125 // re-calculate matrix
126 pde_operator.calculate_system_matrix(system_matrix);
127
128 // initialize Trilinos' AMG
129 amg.initialize(system_matrix, ml_data);
130 }
131
132 void
133 vmult(VectorType & dst, VectorType const & src) const override
134 {
135 amg.vmult(dst, src);
136 }
137
138private:
139 // reference to matrix-free operator
140 Operator const & pde_operator;
141
142 MLData ml_data;
143};

146#ifdef DEAL_II_WITH_PETSC
147/*
148 * Wrapper class for BoomerAMG from Hypre
149 */
150template<typename Operator, typename Number>
151class PreconditionerBoomerAMG : public PreconditionerBase<Number>
152{
153private:
154 typedef LinearAlgebra::distributed::Vector<Number> VectorType;
155
156 typedef PETScWrappers::PreconditionBoomerAMG::AdditionalData BoomerData;
157
158 // subcommunicator; declared before the matrix to ensure that it gets
159 // deleted after the matrix and preconditioner depending on it
160 std::unique_ptr<MPI_Comm, void (*)(MPI_Comm *)> subcommunicator;
161
162public:
163 // distributed sparse system matrix
164 PETScWrappers::MPI::SparseMatrix system_matrix;
165
166 // amg preconditioner for access by PETSc solver
167 PETScWrappers::PreconditionBoomerAMG amg;
168
169 PreconditionerBoomerAMG(Operator const & op, BoomerData boomer_data = BoomerData())
170 : subcommunicator(
171 create_subcommunicator(op.get_matrix_free().get_dof_handler(op.get_dof_index()))),
172 pde_operator(op),
173 boomer_data(boomer_data)
174 {
175 // initialize system matrix
176 pde_operator.init_system_matrix(system_matrix, *subcommunicator);
177
178 calculate_preconditioner();
179 }
180
181 ~PreconditionerBoomerAMG()
182 {
183 if(system_matrix.m() > 0)
184 {
185 PetscErrorCode ierr = VecDestroy(&petsc_vector_dst);
186 AssertThrow(ierr == 0, ExcPETScError(ierr));
187 ierr = VecDestroy(&petsc_vector_src);
188 AssertThrow(ierr == 0, ExcPETScError(ierr));
189 }
190 }
191
192 PETScWrappers::MPI::SparseMatrix const &
193 get_system_matrix()
194 {
195 return system_matrix;
196 }
197
198 void
199 update() override
200 {
201 // clear content of matrix since the next calculate_system_matrix calls
202 // add their result; since we might run this on a sub-communicator, we
203 // skip the processes that do not participate in the matrix and have size
204 // zero
205 if(system_matrix.m() > 0)
206 system_matrix = 0.0;
207
208 calculate_preconditioner();
209 }
210
211 void
212 vmult(VectorType & dst, VectorType const & src) const override
213 {
214 if(system_matrix.m() > 0)
215 apply_petsc_operation(dst,
216 src,
217 petsc_vector_dst,
218 petsc_vector_src,
219 [&](PETScWrappers::VectorBase & petsc_dst,
220 PETScWrappers::VectorBase const & petsc_src) {
221 amg.vmult(petsc_dst, petsc_src);
222 });
223 }
224
225private:
226 void
227 calculate_preconditioner()
228 {
229 // calculate_matrix in case the current MPI rank participates in the PETSc communicator
230 if(system_matrix.m() > 0)
231 {
232 pde_operator.calculate_system_matrix(system_matrix);
233
234 amg.initialize(system_matrix, boomer_data);
235
236 // get vector partitioner
237 LinearAlgebra::distributed::Vector<typename Operator::value_type> vector;
238 pde_operator.initialize_dof_vector(vector);
239 VecCreateMPI(system_matrix.get_mpi_communicator(),
240 vector.get_partitioner()->locally_owned_size(),
241 PETSC_DETERMINE,
242 &petsc_vector_dst);
243 VecCreateMPI(system_matrix.get_mpi_communicator(),
244 vector.get_partitioner()->locally_owned_size(),
245 PETSC_DETERMINE,
246 &petsc_vector_src);
247 }
248 }
249
250 // reference to matrix-free operator
251 Operator const & pde_operator;
252
253 BoomerData boomer_data;
254
255 // PETSc vector objects to avoid re-allocation in every vmult() operation
256 mutable VectorTypePETSc petsc_vector_src;
257 mutable VectorTypePETSc petsc_vector_dst;
258};
259#endif
