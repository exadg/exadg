21 inline double
22 measure_operator_evaluation_time(std::function<void(void)> const & evaluate_operator,
23  unsigned int const degree,
24  unsigned int const n_repetitions_inner,
25  unsigned int const n_repetitions_outer,
26  MPI_Comm const & mpi_comm)
27 {
28  (void)degree;
29 
30  Timer global_timer;
31  global_timer.restart();
32  Utilities::MPI::MinMaxAvg global_time;
33 
34  double wall_time = std::numeric_limits<double>::max();
35 
36  do
37  {
38  for(unsigned int i_outer = 0; i_outer < n_repetitions_outer; ++i_outer)
39  {
40  Timer timer;
41  timer.restart();
42 
43 #ifdef LIKWID_PERFMON
44  LIKWID_MARKER_START(("degree_" + std::to_string(degree)).c_str());
45 #endif
46 
47  // apply matrix-vector product several times
48  for(unsigned int i = 0; i < n_repetitions_inner; ++i)
49  {
50  evaluate_operator();
51  }
52 
53 #ifdef LIKWID_PERFMON
54  LIKWID_MARKER_STOP(("degree_" + std::to_string(degree)).c_str());
55 #endif
56 
57  MPI_Barrier(mpi_comm);
58  Utilities::MPI::MinMaxAvg wall_time_inner =
59  Utilities::MPI::min_max_avg(timer.wall_time(), mpi_comm);
60 
61  wall_time = std::min(wall_time, wall_time_inner.avg / (double)n_repetitions_inner);
62  }
63 
64  global_time = Utilities::MPI::min_max_avg(global_timer.wall_time(), mpi_comm);
65  } while(global_time.avg < 1.0 /*wall time in seconds*/);
66 
67  return wall_time;
68 }
