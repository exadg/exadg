140  template<typename T>
141  void
142  insert_element(std::vector<T> & vector,
143  std::map<std::string, unsigned int> & map,
144  T const & element,
145  std::string const & name)
146  {
147  unsigned int index = vector.size();
148 
149  auto it = map.find(name);
150 
151  // make sure that this element does not already exist
152  if(it == map.end())
153  {
154  map.insert(std::pair<std::string, unsigned int>(name, index));
155  }
156  else
157  {
158  AssertThrow(it == map.end(), ExcMessage("Element already exists. Aborting."));
159  }
160 
161  vector.resize(index + 1);
162  vector.at(index) = element;
163  }

165  unsigned int

167  {
168  auto it = map.find(name);
169 
170  unsigned int index = numbers::invalid_unsigned_int;
171 
172  if(it != map.end())
173  {
174  index = it->second;
175  }
176  else
177  {
178  AssertThrow(it != map.end(), ExcMessage("Could not find element. Aborting."));
179  }
180 
181  return index;
182  }
