99 /*
100  * Calculates the time derivative
101  *
102  * derivative = du/dt = (gamma_0 u^{n+1} - alpha_0 u^{n} - alpha_1 u^{n-1} ... - alpha_{J-1}
103  * u^{n-J+1})/dt
104  */
105 template<typename VectorType>
106 void
107 compute_bdf_time_derivative(VectorType & derivative,
108  VectorType const & solution_np,
109  std::vector<VectorType> const & previous_solutions,
110  BDFTimeIntegratorConstants const & bdf,
111  double const & time_step_size)
112 {
113  derivative.equ(bdf.get_gamma0() / time_step_size, solution_np);
114 
115  for(unsigned int i = 0; i < previous_solutions.size(); ++i)
116  derivative.add(-bdf.get_alpha(i) / time_step_size, previous_solutions[i]);
117 }
