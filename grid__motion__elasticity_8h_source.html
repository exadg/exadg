38template<int dim, typename Number>
39class GridMotionElasticity : public GridMotionBase<dim, Number>
40{
41public:
42 typedef dealii::LinearAlgebra::distributed::Vector<Number> VectorType;
43
47 GridMotionElasticity(std::shared_ptr<dealii::Mapping<dim> const> mapping_undeformed,
48 std::shared_ptr<Structure::Operator<dim, Number>> structure_operator,
49 Structure::Parameters const & structure_parameters)
50 : GridMotionBase<dim, Number>(mapping_undeformed,
51 // extract mapping_degree_moving from elasticity operator
52 structure_operator->get_dof_handler().get_fe().degree,
53 structure_operator->get_dof_handler().get_triangulation()),
54 pde_operator(structure_operator),
55 param(structure_parameters),
56 pcout(std::cout,
57 dealii::Utilities::MPI::this_mpi_process(
58 structure_operator->get_dof_handler().get_communicator()) == 0),
59 iterations({0, {0, 0}})
60 {
61 pde_operator->initialize_dof_vector(displacement);
62 }
63
67 void
68 update(double const time, bool const print_solver_info) override
69 {
70 dealii::Timer timer;
71 timer.restart();
72
73 if(param.large_deformation) // nonlinear problem
74 {
75 VectorType const_vector;
76
77 auto const iter = pde_operator->solve_nonlinear(
78 displacement, const_vector, 0.0 /* no mass term */, time, param.update_preconditioner);
79
80 iterations.first += 1;
81 std::get<0>(iterations.second) += std::get<0>(iter);
82 std::get<1>(iterations.second) += std::get<1>(iter);
83
84 if(print_solver_info)
85 {
86 this->pcout << std::endl << "Solve moving mesh problem (nonlinear elasticity):";
87 print_solver_info_nonlinear(pcout, std::get<0>(iter), std::get<1>(iter), timer.wall_time());
88 }
89 }
90 else // linear problem
91 {
92 // calculate right-hand side vector
93 VectorType rhs;
94 pde_operator->initialize_dof_vector(rhs);
95 pde_operator->compute_rhs_linear(rhs, time);
96
97 auto const iter = pde_operator->solve_linear(displacement, rhs, 0.0 /* no mass term */, time);
98
99 iterations.first += 1;
100 std::get<1>(iterations.second) += iter;
101
102 if(print_solver_info)
103 {
104 this->pcout << std::endl << "Solve moving mesh problem (linear elasticity):";
105 print_solver_info_linear(pcout, iter, timer.wall_time());
106 }
107 }
108
109 this->moving_mapping->initialize_mapping_q_cache(this->mapping_undeformed,
110 displacement,
111 pde_operator->get_dof_handler());
112 }
113
117 void
118 print_iterations() const override
119 {
120 std::vector<std::string> names;
121 std::vector<double> iterations_avg;
122
123 if(param.large_deformation)
124 {
125 names = {"Nonlinear iterations",
126 "Linear iterations (accumulated)",
127 "Linear iterations (per nonlinear it.)"};
128
129 iterations_avg.resize(3);
130 iterations_avg[0] =
131 (double)std::get<0>(iterations.second) / std::max(1., (double)iterations.first);
132 iterations_avg[1] =
133 (double)std::get<1>(iterations.second) / std::max(1., (double)iterations.first);
134 if(iterations_avg[0] > std::numeric_limits<double>::min())
135 iterations_avg[2] = iterations_avg[1] / iterations_avg[0];
136 else
137 iterations_avg[2] = iterations_avg[1];
138 }
139 else // linear
140 {
141 names = {"Linear iterations"};
142 iterations_avg.resize(1);
143 iterations_avg[0] =
144 (double)std::get<1>(iterations.second) / std::max(1., (double)iterations.first);
145 }
146
147 print_list_of_iterations(pcout, names, iterations_avg);
148 }
149
150private:
151 std::shared_ptr<Structure::Operator<dim, Number>> pde_operator;
152
153 Structure::Parameters const & param;
154
155 // store solution of previous time step / iteration so that a good initial
156 // guess is available in the next step, easing convergence or reducing computational
157 // costs by allowing larger tolerances
158 VectorType displacement;
159
160 dealii::ConditionalOStream pcout;
161
162 std::pair<
163 unsigned int /* calls */,
164 std::tuple<unsigned long long, unsigned long long> /* iteration counts {Newton, linear}*/>
165 iterations;
166};
