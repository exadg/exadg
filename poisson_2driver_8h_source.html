54inline std::string
55enum_to_string(OperatorType const enum_type)
56{
57 std::string string_type;

81inline unsigned int
82get_dofs_per_element(std::string const & input_file,
83 unsigned int const dim,
84 unsigned int const degree)
85{
86 std::string spatial_discretization = "DG";
87
88 ParameterHandler prm;
89 prm.enter_subsection("Discretization");
90 prm.add_parameter("SpatialDiscretization",
91 spatial_discretization,
92 "Spatial discretization (CG vs. DG).",
93 Patterns::Selection("CG|DG"),
94 true);
95 prm.leave_subsection();
96
97 prm.parse_input(input_file, "", true, true);
98
99 unsigned int dofs_per_element = 1;
100
101 if(spatial_discretization == "CG")
102 dofs_per_element = Utilities::pow(degree, dim);
103 else if(spatial_discretization == "DG")
104 dofs_per_element = Utilities::pow(degree + 1, dim);
105 else
106 AssertThrow(false, ExcMessage("Not implemented."));
107
108 return dofs_per_element;
109}
