30 // use partial specialization of templates to handle the 2-dimensional case
31 template<typename FEEval>
32 struct CurlCompute<2, FEEval>
33 {
34  static Tensor<1, 2, VectorizedArray<typename FEEval::number_type>>
35  compute(FEEval & fe_eval, const unsigned int q_point)
36  {
37  Tensor<1, 2, VectorizedArray<typename FEEval::number_type>> curl;
38  /*
39  * fe_eval = / phi \ _____\ grad(fe_eval) = / d(phi)/dx1 d(phi)/dx2 \
40  * \ 0 / / \ 0 0 /
41  */
42  Tensor<2, 2, VectorizedArray<typename FEEval::number_type>> temp =
43  fe_eval.get_gradient(q_point);
44  /*
45  * __ / 0 \ / d(phi)/dx2 \
46  * curl = \/ X | 0 | = | - d(phi)/dx1 |
47  * \ phi / \ 0 /
48  */
49  curl[0] = temp[0][1]; // d(phi)/dx2
50  curl[1] = -temp[0][0]; // - d(phi)/dx1
51  return curl;
52  }
53 };
