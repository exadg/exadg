35template<int dim, typename Number>
36void
37my_point_value(dealii::Vector<Number> & result,
38 dealii::Mapping<dim> const & mapping,
39 dealii::DoFHandler<dim> const & dof_handler,
40 dealii::LinearAlgebra::distributed::Vector<Number> const & dof_vector,
41 typename dealii::DoFHandler<dim>::active_cell_iterator const & cell,
42 dealii::Point<dim> const & point_in_ref_coord)
43{
44 Assert(dealii::GeometryInfo<dim>::distance_to_unit_cell(point_in_ref_coord) < 1e-10,
45 dealii::ExcInternalError());
46
47 dealii::Quadrature<dim> const quadrature(
48 dealii::GeometryInfo<dim>::project_to_unit_cell(point_in_ref_coord));
49
50 dealii::FiniteElement<dim> const & fe = dof_handler.get_fe();
51 dealii::FEValues<dim> fe_values(mapping, fe, quadrature, dealii::update_values);
52 fe_values.reinit(cell);
53
54 // then use this to get the values of the given fe_function at this point
55 std::vector<dealii::Vector<Number>> solution_vector(1, dealii::Vector<Number>(fe.n_components()));
56 fe_values.get_function_values(dof_vector, solution_vector);
57 result = solution_vector[0];
58}

60template<int dim, typename Number>
61void
62evaluate_scalar_quantity_in_point(
63 Number & solution_value,
64 dealii::DoFHandler<dim> const & dof_handler,
65 dealii::Mapping<dim> const & mapping,
66 dealii::LinearAlgebra::distributed::Vector<Number> const & numerical_solution,
67 dealii::Point<dim> const & point,
68 MPI_Comm const & mpi_comm,
69 double const tolerance = 1.e-10)
70{
71 typedef std::pair<typename dealii::DoFHandler<dim>::active_cell_iterator, dealii::Point<dim>>
72 Pair;
73
74 std::vector<Pair> adjacent_cells =
75 dealii::GridTools::find_all_active_cells_around_point(mapping, dof_handler, point, tolerance);
76
77 // processor local variables: initialize with zeros since we add values to these variables
78 unsigned int counter = 0;
79 solution_value = 0.0;
80
81 // loop over all adjacent cells
82 for(auto cell : adjacent_cells)
83 {
84 // go on only if cell is owned by the processor
85 if(cell.first->is_locally_owned())
86 {
87 dealii::Vector<Number> value(1);
88 my_point_value(value, mapping, dof_handler, numerical_solution, cell.first, cell.second);
89
90 solution_value += value(0);
91 ++counter;
92 }
93 }
94
95 // parallel computations: add results of all processors and calculate mean value
96 counter = dealii::Utilities::MPI::sum(counter, mpi_comm);
97 AssertThrow(counter > 0, dealii::ExcMessage("No points found."));
98
99 solution_value = dealii::Utilities::MPI::sum(solution_value, mpi_comm);
100 solution_value /= (double)counter;
101}

146 solution_value /= (double)counter;

149/*
150 * For a given vector of adjacent cells and points in reference coordinates, determine
151 * and return the dof_indices and shape_values to be used later for interpolation of the
152 * solution.
153 */
154template<int dim, typename Number>
155std::vector<std::pair<std::vector<dealii::types::global_dof_index>, std::vector<Number>>>
156get_dof_indices_and_shape_values(
157 std::vector<std::pair<typename dealii::Triangulation<dim>::active_cell_iterator,
158 dealii::Point<dim>>> const & adjacent_cells,
159 dealii::DoFHandler<dim> const & dof_handler,
160 dealii::Mapping<dim> const & mapping,
161 dealii::LinearAlgebra::distributed::Vector<Number> const & solution)
162{
163 // fill dof_indices_and_shape_values
164 std::vector<std::pair<std::vector<dealii::types::global_dof_index>, std::vector<Number>>>
165 dof_indices_and_shape_values;
166
167 for(auto cell_tria : adjacent_cells)
168 {
169 // transform Triangulation::active_cell_iterator into dealii::DoFHandler::active_cell_iterator,
170 // see constructor of DoFCellAccessor
171 typename dealii::DoFHandler<dim>::active_cell_iterator cell_dof = {
172 &dof_handler.get_triangulation(),
173 cell_tria.first->level(),
174 cell_tria.first->index(),
175 &dof_handler};
176
177 typedef std::pair<typename dealii::DoFHandler<dim>::active_cell_iterator, dealii::Point<dim>>
178 PairDof;
179 PairDof cell_and_point(cell_dof, cell_tria.second);
180
181 // go on only if cell is owned by the processor
182 if(cell_and_point.first->is_locally_owned())
183 {
184 const dealii::Quadrature<dim> quadrature(
185 dealii::GeometryInfo<dim>::project_to_unit_cell(cell_and_point.second));
186
187 const dealii::FiniteElement<dim> & fe = dof_handler.get_fe();
188 dealii::FEValues<dim> fe_values(mapping, fe, quadrature, dealii::update_values);
189 fe_values.reinit(cell_and_point.first);
190 std::vector<dealii::types::global_dof_index> dof_indices(fe.dofs_per_cell);
191 cell_and_point.first->get_dof_indices(dof_indices);
192
193 std::vector<dealii::types::global_dof_index> dof_indices_global(fe.dofs_per_cell);
194 for(unsigned int i = 0; i < fe.dofs_per_cell; ++i)
195 dof_indices_global[i] = solution.get_partitioner()->global_to_local(dof_indices[i]);
196
197 std::vector<Number> fe_shape_values(fe.dofs_per_cell);
198 for(unsigned int i = 0; i < fe.dofs_per_cell; ++i)
199 fe_shape_values[i] = fe_values.shape_value(i, 0);
200
201 dof_indices_and_shape_values.emplace_back(dof_indices_global, fe_shape_values);
202 }
203 }
204
205 return dof_indices_and_shape_values;
206}

208/*
209 * Interpolate solution from dof vector by using precomputed shape function values.
210 */
211template<int rank, int dim, typename Number>
212struct Interpolator
213{
214 static inline DEAL_II_ALWAYS_INLINE //
215 dealii::Tensor<rank, dim, Number>
216 value(dealii::DoFHandler<dim> const & dof_handler,
217 dealii::LinearAlgebra::distributed::Vector<Number> const & solution,
218 std::vector<dealii::types::global_dof_index> const & dof_indices,
219 std::vector<Number> const & fe_shape_values)
220 {
221 (void)dof_handler;
222 (void)solution;
223 (void)dof_indices;
224 (void)fe_shape_values;
225
226 AssertThrow(false, dealii::ExcMessage("not implemented."));
227
228 dealii::Tensor<rank, dim, Number> result;
229 return result;
230 }
231};

233/*
234 * The quantity to be evaluated is of type dealii::Tensor<0,dim,Number>.
235 */
236template<int dim, typename Number>
237struct Interpolator<0, dim, Number>
238{
239 static inline DEAL_II_ALWAYS_INLINE //
240 dealii::Tensor<0, dim, Number>
241 value(dealii::DoFHandler<dim> const & dof_handler,
242 dealii::LinearAlgebra::distributed::Vector<Number> const & solution,
243 std::vector<dealii::types::global_dof_index> const & dof_indices,
244 std::vector<Number> const & fe_shape_values)
245 {
246 Assert(dof_handler.get_fe().dofs_per_cell == fe_shape_values.size(),
247 dealii::ExcMessage("Vector fe_shape_values has wrong size."));
248
249 Number result = Number(0.0);
250
251 dealii::FiniteElement<dim> const & fe = dof_handler.get_fe();
252 for(unsigned int i = 0; i < fe.dofs_per_cell; ++i)
253 result += solution.local_element(dof_indices[i]) * fe_shape_values[i];
254
255 dealii::Tensor<0, dim, Number> result_tensor = result;
256
257 return result_tensor;
258 }
259};

261/*
262 * The quantity to be evaluated is of type dealii::Tensor<1, dim, Number>.
263 */
264template<int dim, typename Number>
265struct Interpolator<1, dim, Number>
266{
267 static inline DEAL_II_ALWAYS_INLINE //
268 dealii::Tensor<1, dim, Number>
269 value(dealii::DoFHandler<dim> const & dof_handler,
270 dealii::LinearAlgebra::distributed::Vector<Number> const & solution,
271 std::vector<dealii::types::global_dof_index> const & dof_indices,
272 std::vector<Number> const & fe_shape_values)
273 {
274 Assert(dof_handler.get_fe().dofs_per_cell == fe_shape_values.size(),
275 dealii::ExcMessage("Vector fe_shape_values has wrong size."));
276
277 dealii::Tensor<1, dim, Number> result;
278
279 dealii::FiniteElement<dim> const & fe = dof_handler.get_fe();
280 for(unsigned int i = 0; i < fe.dofs_per_cell; ++i)
281 result[fe.system_to_component_index(i).first] +=
282 solution.local_element(dof_indices[i]) * fe_shape_values[i];
283
284 return result;
285 }
