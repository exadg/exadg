23 /*
24  * This function calculates the penalty parameter of the interior
25  * penalty method for each cell.
26  */
27 template<int dim, typename Number>
28 void
29 calculate_penalty_parameter(AlignedVector<VectorizedArray<Number>> & array_penalty_parameter,
30  MatrixFree<dim, Number> const & matrix_free,
31  unsigned int const dof_index = 0)
32 {
33  unsigned int n_cells = matrix_free.n_cell_batches() + matrix_free.n_ghost_cell_batches();
34  array_penalty_parameter.resize(n_cells);
35 
36  Mapping<dim> const & mapping = *matrix_free.get_mapping_info().mapping;
37  FiniteElement<dim> const & fe = matrix_free.get_dof_handler(dof_index).get_fe();
38  unsigned int const degree = fe.degree;
39 
40  QGauss<dim> quadrature(degree + 1);
41  FEValues<dim> fe_values(mapping, fe, quadrature, update_JxW_values);
42 
43  QGauss<dim - 1> face_quadrature(degree + 1);
44  FEFaceValues<dim> fe_face_values(mapping, fe, face_quadrature, update_JxW_values);
45 
46  for(unsigned int i = 0; i < n_cells; ++i)
47  {
48  for(unsigned int v = 0; v < matrix_free.n_active_entries_per_cell_batch(i); ++v)
49  {
50  typename DoFHandler<dim>::cell_iterator cell = matrix_free.get_cell_iterator(i, v, dof_index);
51  fe_values.reinit(cell);
52 
53  // calculate cell volume
54  Number volume = 0;
55  for(unsigned int q = 0; q < quadrature.size(); ++q)
56  {
57  volume += fe_values.JxW(q);
58  }
59 
60  // calculate surface area
61  Number surface_area = 0;
62  for(unsigned int f = 0; f < GeometryInfo<dim>::faces_per_cell; ++f)
63  {
64  fe_face_values.reinit(cell, f);
65  Number const factor = (cell->at_boundary(f) && !cell->has_periodic_neighbor(f)) ? 1. : 0.5;
66  for(unsigned int q = 0; q < face_quadrature.size(); ++q)
67  {
68  surface_area += fe_face_values.JxW(q) * factor;
69  }
70  }
71 
72  array_penalty_parameter[i][v] = surface_area / volume;
73  }
74  }
75 }

77 /*
78  * This function returns the penalty factor of the interior penalty method
79  * for quadrilateral/hexahedral elements for a given polynomial degree of
80  * the shape functions and a specified penalty factor (scaling factor).
81  */
82 
83 template<typename Number>
84 Number
85 get_penalty_factor(unsigned int const degree, Number const factor = 1.0)
86 {
87  return factor * (degree + 1.0) * (degree + 1.0);
88 }
