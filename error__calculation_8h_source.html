37template<int dim>
38struct ErrorCalculationData
39{
40 ErrorCalculationData()
41 : calculate_relative_errors(true),
42 calculate_H1_seminorm_error(false),
43 write_errors_to_file(false),
44 directory("output/"),
45 name("all fields")
46 {
47 }
48
49 void
50 print(dealii::ConditionalOStream & pcout, bool unsteady)
51 {
52 print_parameter(pcout, "Error calculation", unsteady == true and analytical_solution);
53 if(unsteady == true and time_control_data.is_active)
54 {
55 print(pcout, unsteady, time_control_data);
56 print_parameter(pcout, "Calculate relative errors", calculate_relative_errors);
57 print_parameter(pcout, "Calculate H1-seminorm error", calculate_H1_seminorm_error);
58 print_parameter(pcout, "Write errors to file", write_errors_to_file);
59 if(write_errors_to_file)
60 print_parameter(pcout, "Directory", directory);
61 print_parameter(pcout, "Name", name);
62 }
63 }
64
65 std::shared_ptr<dealii::Function<dim>> analytical_solution;
66
67 // relative or absolute errors?
68 // If calculate_relative_errors == false, this implies that absolute errors are calculated
69 bool calculate_relative_errors;
70
71 // by default, only the L2-error is computed. Other norms have to be explicitly specified by the
72 // user.
73 bool calculate_H1_seminorm_error;
74
75 // data used to control the output
76 TimeControlData time_control_data;
77
78 // write errors to file?
79 bool write_errors_to_file;
80
81 // directory and name (used as filename and as identifier for screen output)
82 std::string directory;
83 std::string name;
84};
