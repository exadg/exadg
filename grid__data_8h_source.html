32/*
33 * Triangulation type

35enum class TriangulationType

37 Serial,

42/*

47 Hypercube,

51/*
52 * Partitioning type (relevant for fully-distributed triangulation)

56 Metis,

60struct GridData
61{
62 GridData()
63 : triangulation_type(TriangulationType::Distributed),
64 element_type(ElementType::Hypercube),
65 partitioning_type(PartitioningType::Metis),
66 n_refine_global(0),
67 file_name(),
68 create_coarse_triangulations(false)
69 {
70 }
71
72 void
73 check() const
74 {
75 }
76
77 void
78 print(dealii::ConditionalOStream const & pcout) const
79 {
80 print_parameter(pcout, "Triangulation type", triangulation_type);
81
82 print_parameter(pcout, "Element type", element_type);
83
84 if(triangulation_type == TriangulationType::FullyDistributed)
85 print_parameter(pcout, "Partitioning type (fully-distributed)", partitioning_type);
86
87 print_parameter(pcout, "Number of global refinements", n_refine_global);
88
89 if(not file_name.empty())
90 print_parameter(pcout, "Grid file name", file_name);
91
92 print_parameter(pcout, "Create coarse triangulations", create_coarse_triangulations);
93 }
94
95 TriangulationType triangulation_type;
96
97 ElementType element_type;
98
99 // only relevant for TriangulationType::FullyDistributed
100 PartitioningType partitioning_type;
101
102 unsigned int n_refine_global;
103
104 // path to a grid file
105 // the filename needs to include a proper filename ending/extension so that we can internally
106 // deduce the correct type of the file format
107 std::string file_name;
108
109 // In case of a hypercube mesh that is globally refined, i.e. without hanging nodes, the fine
110 // triangulation can be used for all multigrid h-levels without the need to create coarse
111 // triangulations explicitly. Hence, this parameter is typically set to false for globally-refined
112 // hypercube meshes.
113 // Nevertheless, it is possible to set this parameter to true for globally-refined hypercube
114 // meshes. In that case, the coarse triangulations are created explicitly for use in
115 // h-multigrid methods.
116 // This parameter needs to be set to true if one wants to use h-multigrid methods for
117 // locally-refined hypercube meshes or non-hypercube meshes.
118 bool create_coarse_triangulations;
119};
120
121} // namespace ExaDG
